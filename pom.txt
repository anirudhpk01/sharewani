            list(chunk), key_prefix=redis_prefix
        )
        writes: dict[str, dict] = {}

        async def handle(cid: str):
            # unchanged
            ...

        tasks = [asyncio.create_task(handle(cid)) for cid in chunk]
        if tasks:

        if writes:
