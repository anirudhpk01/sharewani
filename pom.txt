"""
app/core/connectionpool.py

Full replacement implementing the new low-latency CDC pipeline.

ARCHITECTURE
============

Hot path per CDC event (in the watcher loop, ZERO I/O):
  - Extract conversationId + minimal doc fields from fullDocument
  - Append to in-memory pending buffer: {cid: {doc_id_str: slim_doc}}
  - Mark conversationId as dirty, signal the consumer

Per batch (single coalescing consumer loop, no fixed sleep):
  1. Atomically swap out current pending buffer + dirty cids set.
     If empty, briefly wait (event.wait with short timeout) for new events.
  2. ONE pipelined Redis read: HGETALL doccache:{cid} for all cids + MGET aggregated payloads.
  3. Apply per-doc out-of-order fence using lastModifiedDateTime (`_lm`).
  4. Concurrent aggregation per cid (CPU-bound, gated by agg_sem).
  5. ONE pipelined Redis write: HSET doc-cache fields + SET aggregated payloads
     + DEL doc-cache for any conversation now closed.
  6. Enqueue the totals delta event for the totals consumer.
  7. Loop immediately. The processing time of this iteration IS the
     accumulation window for the next iteration — no fixed sleep, no stalls.

This Nagle-style coalescing means:
  - Under low load: each batch processes immediately (latency ≈ Redis RTT).
  - Under high load: natural batching, ~10-20 round-trips/sec total.
"""

def _parse_date_value(raw_value: str) -> datetime | None:
    if not raw_value:
        return None
    value = raw_value.strip()
    if not value:
        return None
    for fmt in ("%Y-%m-%d", "%d/%m/%Y", "%m/%d/%Y"):
        try:
            dt = datetime.strptime(value, fmt)
            return dt.replace(tzinfo=timezone.utc)
        except Exception:
            pass
    try:
        iso_value = value[:-1] + "+00:00" if value.endswith("Z") else value
        dt = datetime.fromisoformat(iso_value)
        return dt if dt.tzinfo else dt.replace(tzinfo=timezone.utc)
    except Exception:
        return None


def _build_datetime_filter(key_name: str, raw_value):
    if not key_name or "date" not in key_name.lower():
        return None
    parsed: datetime | None = None
    date_only = False
    if isinstance(raw_value, datetime):
        parsed = raw_value if raw_value.tzinfo else raw_value.replace(tzinfo=timezone.utc)
        date_only = parsed.timetz() == time(0, 0)
    elif isinstance(raw_value, str):
        parsed = _parse_date_value(raw_value)
        date_only = "T" not in raw_value and len(raw_value.strip()) <= 10
    if not parsed:
        return None
    start = parsed.replace(hour=0, minute=0, second=0, microsecond=0) if date_only else parsed
    now_utc = datetime.now(timezone.utc)
    return {"$gte": start, "$lte": now_utc}


def _coerce_cdc_datetime(raw_value) -> datetime | None:
    if raw_value is None:
        return None
    if isinstance(raw_value, datetime):
        return raw_value if raw_value.tzinfo else raw_value.replace(tzinfo=timezone.utc)
    if hasattr(raw_value, "as_datetime"):
        try:
            parsed = raw_value.as_datetime()
            return parsed if parsed.tzinfo else parsed.replace(tzinfo=timezone.utc)
        except Exception:
            pass
    if hasattr(raw_value, "time"):
        try:
            return datetime.fromtimestamp(int(raw_value.time), tz=timezone.utc)
        except Exception:
            pass
    if isinstance(raw_value, str):
        return _parse_date_value(raw_value)
    return None


def _parse_iso_safe(value) -> datetime | None:
    if value is None:
        return None
    if isinstance(value, datetime):
        return value if value.tzinfo else value.replace(tzinfo=timezone.utc)
    try:
        s = str(value).strip()
        if not s:
            return None
        if s.endswith("Z"):
            s = s[:-1] + "+00:00"
        dt = datetime.fromisoformat(s)
        return dt if dt.tzinfo else dt.replace(tzinfo=timezone.utc)
    except Exception:
        return None

def _extract_conversation_id(change: dict) -> str | None:
    full_doc = change.get("fullDocument") if isinstance(change, dict) else None
    if isinstance(full_doc, dict) and full_doc.get("conversationId"):
        return str(full_doc["conversationId"])
    doc_key = change.get("documentKey") if isinstance(change, dict) else None
    if isinstance(doc_key, dict) and doc_key.get("conversationId"):
        return str(doc_key["conversationId"])
    return None


def _extract_doc_id_str(change: dict) -> str | None:
    full_doc = change.get("fullDocument") if isinstance(change, dict) else None
    if isinstance(full_doc, dict) and full_doc.get("_id") is not None:
        return str(full_doc["_id"])
    doc_key = change.get("documentKey") if isinstance(change, dict) else None
    if isinstance(doc_key, dict) and doc_key.get("_id") is not None:
        return str(doc_key["_id"])
    return None


def _slim_doc_for_cache(full_doc: dict, doc_id_str: str, fallback_lm: str | None) -> dict:
    """Build a minimal cache entry. Keeps only fields read by aggregation +
    an `_lm` sentinel used for the out-of-order fence.
    """
    slim: dict = {}
    for field in _CDC_CACHED_FIELDS:
        if field == "_id":
            slim["_id"] = doc_id_str
            continue
        value = full_doc.get(field)
        if value is None:
            continue
        if isinstance(value, datetime):
            slim[field] = (
                value if value.tzinfo else value.replace(tzinfo=timezone.utc)
            ).isoformat()
        else:
            slim[field] = value
    last_modified = full_doc.get("lastModifiedDateTime") or fallback_lm
    if isinstance(last_modified, datetime):
        last_modified = (
            last_modified if last_modified.tzinfo else last_modified.replace(tzinfo=timezone.utc)
        ).isoformat()
    slim["_lm"] = last_modified or datetime.now(timezone.utc).isoformat()
    return slim


def _is_payload_closed(payload: dict | None) -> bool:
    """Detect 'no more CDC events will arrive for this conversation'."""
    if not isinstance(payload, dict):
        return False
    if int(payload.get("conversation_closed_ind") or 0) == 1:
        return True
    state = str(payload.get("conversation_state") or "").upper()
    return state in _CLOSED_STATES

def _extract_cdc_event_metadata(change, captured_at: datetime | None = None) -> dict[str, object]:
    normalized_captured_at = captured_at or datetime.now(timezone.utc)
    change_dict = change if isinstance(change, dict) else {}
    full_doc = change_dict.get("fullDocument") if isinstance(change_dict.get("fullDocument"), dict) else {}
    document_key = change_dict.get("documentKey") if isinstance(change_dict.get("documentKey"), dict) else {}

    cluster_time = _coerce_cdc_datetime(change_dict.get("clusterTime"))
    wall_time = _coerce_cdc_datetime(change_dict.get("wallTime"))
    last_modified_at = _coerce_cdc_datetime(full_doc.get("lastModifiedDateTime"))

    event_time = wall_time or cluster_time or last_modified_at
    event_time_source = (
        "wallTime" if wall_time
        else "clusterTime" if cluster_time
        else "lastModifiedDateTime" if last_modified_at
        else None
    )
    capture_lag_ms = None
    if event_time is not None:
        capture_lag_ms = max(0, int((normalized_captured_at - event_time).total_seconds() * 1000))

    return {
        "cid": full_doc.get("conversationId") or document_key.get("conversationId"),
        "document_id": document_key.get("_id") or full_doc.get("_id"),
        "operation_type": change_dict.get("operationType"),
        "event_time": event_time.isoformat() if event_time else None,
        "event_time_source": event_time_source,
        "capture_lag_ms": capture_lag_ms,
        "capture_lag_source": event_time_source,
    }

def _is_invalid_resume_token_error(exc: Exception) -> bool:
    message = str(exc).lower()
    markers = ("resume", "history lost", "changestreamhistorylost",
               "non-resumable", "cannot resume", "resume point", "token")
    return any(marker in message for marker in markers)


def _resolve_mongo_read_preference(preference_name: str | None, default: str = "primary"):
    normalized = str(preference_name or default).strip().lower()
    if normalized == "secondarypreferred":
        return SecondaryPreferred()
    return Primary()

def _normalize_totals_delta_event(
    previous_states: dict[str, object] | None,
    current_states: dict[str, object] | None,
    deleted_states: dict[str, object] | None,
) -> dict[str, dict[str, object]]:
    return {
        "previous_states": dict(previous_states or {}),
        "current_states": dict(current_states or {}),
        "deleted_states": dict(deleted_states or {}),
    }


def _merge_totals_delta_events(
    events: list[dict[str, dict[str, object]]],
) -> tuple[dict[str, object], dict[str, object], dict[str, object]]:
    merged: dict[str, dict[str, object]] = {}
    for event in events:
        previous_states = event.get("previous_states") or {}
        current_states = event.get("current_states") or {}
        deleted_states = event.get("deleted_states") or {}
        conversation_ids = set(previous_states) | set(current_states) | set(deleted_states)
        for cid in conversation_ids:
            entry = merged.setdefault(cid, {"previous": None, "current": None, "deleted": False})
            if entry["previous"] is None and cid in previous_states:
                entry["previous"] = previous_states.get(cid)
            if cid in current_states:
                entry["current"] = current_states.get(cid)
                entry["deleted"] = False
            if cid in deleted_states:
                entry["current"] = None
                entry["deleted"] = True

    previous = {cid: entry["previous"] for cid, entry in merged.items() if entry.get("previous") is not None}
    current = {cid: entry["current"] for cid, entry in merged.items() if entry.get("current") is not None}
    deleted = {cid: None for cid, entry in merged.items() if entry.get("deleted")}
    return previous, current, deleted

# =============================================================================
# MongoDBConnectionPool — singleton client + CDC orchestrator
# =============================================================================

class MongoDBConnectionPool:
    _client = None
    _cdc_client = None
    _config = None
    _lock = asyncio.Lock()

    # ------------------------------------------------------------------
    # Lifecycle
    # ------------------------------------------------------------------

    @classmethod
    async def init_pool(cls, config=None):
        async with cls._lock:
            if cls._client is None:
                if config is None:
                    config = ConfigReader().get_mongo_config()
                cls._config = config
                cls._client = AsyncIOMotorClient(config["uri"], **cls._build_query_client_kwargs(config))
                cls._cdc_client = AsyncIOMotorClient(config["uri"], **cls._build_cdc_client_kwargs(config))

    @classmethod
    async def close_pool(cls):
        if cls._client is not None:
            cls._client.close()
            cls._client = None
        if cls._cdc_client is not None:
            cls._cdc_client.close()
            cls._cdc_client = None

    @classmethod
    def _build_query_client_kwargs(cls, config: dict) -> dict:
        configured_max = max(1, int(config["max_pool_size"]))
        configured_min = max(0, int(config["min_pool_size"]))
        recommended_max = max(_estimate_required_query_pool_size(), configured_max)
        recommended_min = min(configured_min, recommended_max)
        read_preference = _resolve_mongo_read_preference(
            os.getenv("MONGO_QUERY_READ_PREFERENCE"), default="primary"
        )
        return {
            "maxPoolSize": recommended_max,
            "minPoolSize": recommended_min,
            "maxIdleTimeMS": config["idle_timeout_ms"],
            "appname": "prompt_manager_query",
            "read_preference": read_preference,
        }

    @classmethod
    def _build_cdc_client_kwargs(cls, config: dict) -> dict:
        read_preference = _resolve_mongo_read_preference(
            os.getenv("MONGO_CDC_READ_PREFERENCE"), default="secondaryPreferred"
        )
        return {
            "maxPoolSize": 1,
            "minPoolSize": 1,
            "maxIdleTimeMS": config["idle_timeout_ms"],
            "appname": "prompt_manager_cdc",
            "read_preference": read_preference,
        }

    @classmethod
    async def get_client(cls, *, client_role: str = "query"):
        if cls._client is None:
            raise RuntimeError("MongoDBConnectionPool not initialized. Call init_pool() at startup.")
        if client_role == "cdc":
            if cls._cdc_client is None:
                raise RuntimeError("MongoDB CDC client not initialized. Call init_pool() at startup.")
            return cls._cdc_client
        return cls._client

    @classmethod
    async def get_cdc_client(cls):
        return await cls.get_client(client_role="cdc")

    @classmethod
    async def get_database_collection(cls, dbname, collectionname, *, client_role: str = "query"):
        client = await cls.get_client(client_role=client_role)
        return client[dbname][collectionname]

    @staticmethod
    def serialize(document):
        if isinstance(document, list):
            return [MongoDBConnectionPool.serialize(doc) for doc in document]
        if isinstance(document, dict):
            doc = dict(document)
            doc.pop("_id", None)
            for key, value in doc.items():
                if isinstance(value, datetime):
                    doc[key] = value.isoformat()
                elif isinstance(value, dict):
                    doc[key] = MongoDBConnectionPool.serialize(value)
                elif isinstance(value, list):
                    doc[key] = MongoDBConnectionPool.serialize(value)
            return doc
        return document

    # ------------------------------------------------------------------
    # API-side document fetches (unchanged behaviour)
    # ------------------------------------------------------------------

    @classmethod
    async def get_documents(cls, request: Request, dbname, collectionname, key, key_name):
        logger = get_logger_from_request(request)
        try:
            collection = await cls.get_database_collection(dbname, collectionname)
            range_filter = _build_datetime_filter(key_name, key)
            if range_filter:
                filter_query = {key_name: range_filter}
                start_dt = range_filter.get("$gte")
                end_dt = range_filter.get("$lte")
                if isinstance(start_dt, datetime) and isinstance(end_dt, datetime):
                    filter_query = {
                        "$or": [
                            {key_name: range_filter},
                            {key_name: {"$gte": start_dt.isoformat(), "$lte": end_dt.isoformat()}},
                        ]
                    }
                await logger.info(
                    f"Applying range filter on {key_name}: {range_filter.get('$gte')} to {range_filter.get('$lte')}"
                )
            else:
                filter_query = {key_name: key}
            cursor = collection.find(filter_query, {"_id": 0})
            documents = []
            async for doc in cursor:
                documents.append(doc)
            return cls.serialize(documents) if documents else "No data Available"
        except Exception as e:
            await logger.error(f"Failed to retrieve documents from collection '{collectionname}': {e}")
            return JSONResponse(
                status_code=500,
                content={"error": INTERNAL_SERVER_ERR, "message": INTERNAL_SERVER_MSG, "statusCodeValue": 500},
            )

    @classmethod
    async def get_documents_grouped_by_conversation_ids(
        cls,
        request: Request,
        dbname,
        collectionname,
        conversation_ids: list[str],
    ) -> dict[str, list[dict]]:
        """Used by one-time-load + admin paths only — NOT by the CDC hot path."""
        if not conversation_ids:
            return {}
        logger = get_logger_from_request(request)
        try:
            collection = await cls.get_database_collection(dbname, collectionname)
            projection = dict(_CDC_CONVERSATION_FETCH_PROJECTION)
            fetch_batch_size = int(os.getenv("CDC_FETCH_CURSOR_BATCH_SIZE", "10000"))
            mongo_query = with_allowed_customer_id_filter(
                {"conversationId": {"$in": list(conversation_ids)}}
            )
            cursor = (
                collection.find(mongo_query, projection)
                .sort([("conversationId", 1), ("createdDateTime", 1)])
                .batch_size(fetch_batch_size)
            )
            grouped: dict[str, list[dict]] = {cid: [] for cid in conversation_ids}
            async for raw in cursor:
                if not isinstance(raw, dict):
                    continue
                doc_id_value = raw.get("_id")
                doc = cls.serialize(raw)
                if isinstance(doc, dict):
                    if doc_id_value is not None and "_id" not in doc:
                        doc["_id"] = str(doc_id_value)
                    cid = doc.get("conversationId")
                    if cid:
                        grouped.setdefault(cid, []).append(doc)
            return grouped
        except Exception as e:
            await logger.error(
                f"Failed to retrieve grouped conversation documents from collection '{collectionname}': {e}"
            )
            return {}

    # ------------------------------------------------------------------
    # CDC runtime construction
    # ------------------------------------------------------------------

    @classmethod
    def _resolve_cdc_target_db(cls, dbname: str | None) -> str | None:
        target_db = dbname or (cls._config.get("database_name") if cls._config else None)
        if target_db:
            return target_db
        config = ConfigReader().get_mongo_config()
        return config.get("database_name")

    @classmethod
    async def _build_cdc_runtime(
        cls,
        app,
        dbname: str | None,
        collectionname: str,
        redis_prefix: str,
        max_concurrency: int,
        resume_token=None,
        restart_count: int = 0,
        restart_reason: str | None = None,
    ) -> dict:
        logger = getattr(app.state, "logger", None)
        target_db = cls._resolve_cdc_target_db(dbname)
        cdc_collection = await cls.get_database_collection(target_db, collectionname, client_role="cdc")

        from app.core.redis_connect import AWSRedis
        from app.service.realtime_metrics import RealtimeMetrics

        # Coalescing-loop tuning.
        # max_idle_wait: when the buffer is empty, how long to wait for new events
        # before doing a no-op iteration. Smaller = faster reaction.
        max_idle_wait_ms = int(os.getenv("CDC_IDLE_WAIT_MS", "20"))
        # Cap on cids per iteration (prevents giant pipelines on startup backlog).
        max_batch_cids = max(1, int(os.getenv("CDC_MAX_BATCH_CIDS", "2000")))
        totals_batch_window_ms = int(os.getenv("CDC_TOTALS_BATCH_WINDOW_MS", "300"))

        redis_client = AWSRedis()
        metrics = RealtimeMetrics(
            redis_client=redis_client,
            logger=logger,
            redis_key_prefix=redis_prefix,
        )

        return {
            "logger": logger,
            "target_db": target_db,
            "collection": cdc_collection,
            "collectionname": collectionname,
            "redis_prefix": redis_prefix,
            "redis_client": redis_client,
            "metrics": metrics,
            "request_proxy": _RequestProxy(app),

            # In-memory pending state.
            "pending_buffer": {},        # {cid: {doc_id_str: slim_doc}}
            "dirty_cids": set(),
            "in_flight_cids": set(),
            "buffer_lock": asyncio.Lock(),
            "wakeup_event": asyncio.Event(),

            # Coalescing parameters.
            "max_idle_wait_ms": max_idle_wait_ms,
            "max_idle_wait": max(0.005, max_idle_wait_ms / 1000),
            "max_batch_cids": max_batch_cids,
            "totals_batch_window_ms": totals_batch_window_ms,
            "totals_batch_window": max(0.05, totals_batch_window_ms / 1000),

            "agg_sem": asyncio.Semaphore(max(1, max_concurrency)),
            "totals_queue": asyncio.Queue(),

            # Stream / observability state.
            "resume_token": resume_token,
            "cdc_seen": 0,
            "cdc_stream_started_at": datetime.now(timezone.utc),
            "cdc_last_event_received_at": None,
            "cdc_last_resume_token_seen": str(resume_token) if resume_token is not None else None,
            "cdc_restarts_total": restart_count,
            "cdc_last_restart_reason": restart_reason,
            "cdc_last_capture_lag_ms": None,
            "cdc_max_capture_lag_ms": 0,
            "cdc_last_capture_lag_source": None,
            "cdc_last_warning_at": None,
            "cdc_last_warned_capture_lag_ms": 0,
            "cdc_filtered_changes_total": 0,
            "cdc_filtered_changes_by_reason": {},
            "cdc_processing_failures_total": 0,
            "cdc_processing_failures_by_reason": {},
            "cdc_batches_processed": 0,
            "cdc_totals_enqueue_total": 0,
            "cdc_totals_batches_processed": 0,
            "cdc_filter_criteria": describe_cdc_filter_criteria(),
            "cdc_client_read_preference": type(getattr((await cls.get_cdc_client()), "read_preference", None)).__name__,
        }

    # ------------------------------------------------------------------
    # CDC HOT PATH — watcher (no I/O) → pending buffer
    # ------------------------------------------------------------------

    @classmethod
    async def _handle_cdc_change(cls, runtime: dict, change) -> None:
        """Hot path. Called once per CDC event. NO Redis, NO Mongo."""
        await wait_until_processing_unpaused(runtime["request_proxy"].app)

        # Customer / op filter (preserved exactly from before).
        should_enqueue, reason, _ = evaluate_cdc_change(change)
        if not should_enqueue:
            runtime["cdc_filtered_changes_total"] = int(runtime.get("cdc_filtered_changes_total", 0) or 0) + 1
            _increment_reason_counter(runtime, "cdc_filtered_changes_by_reason", reason)
            return

        conversation_id = _extract_conversation_id(change)
        if not conversation_id:
            runtime["cdc_filtered_changes_total"] = int(runtime.get("cdc_filtered_changes_total", 0) or 0) + 1
            _increment_reason_counter(runtime, "cdc_filtered_changes_by_reason", "conversation_id_missing")
            return

        full_doc = change.get("fullDocument") if isinstance(change, dict) else None
        doc_id_str = _extract_doc_id_str(change)

        # Lag tracking.
        captured_at = runtime.get("cdc_last_event_received_at") or datetime.now(timezone.utc)
        metadata = _extract_cdc_event_metadata(change, captured_at=captured_at)
        runtime["cdc_last_capture_lag_ms"] = metadata.get("capture_lag_ms")
        runtime["cdc_last_capture_lag_source"] = metadata.get("capture_lag_source")
        capture_lag_ms = metadata.get("capture_lag_ms")
        if isinstance(capture_lag_ms, int):
            runtime["cdc_max_capture_lag_ms"] = max(
                int(runtime.get("cdc_max_capture_lag_ms", 0) or 0), capture_lag_ms
            )
            warn_threshold = _cdc_lag_warning_threshold_ms()
            if capture_lag_ms >= warn_threshold:
                last_warning_at = runtime.get("cdc_last_warning_at")
                last_warned_lag = int(runtime.get("cdc_last_warned_capture_lag_ms", 0) or 0)
                should_warn = (
                    last_warning_at is None
                    or (captured_at - last_warning_at).total_seconds() >= 60
                    or capture_lag_ms > last_warned_lag
                )
                if should_warn:
                    runtime["cdc_last_warning_at"] = captured_at
                    runtime["cdc_last_warned_capture_lag_ms"] = capture_lag_ms
                    logger = runtime.get("logger")
                    await _safe_async_log(
                        logger.warning if logger else None,
                        f"CDC capture lag warning cid={metadata.get('cid')} "
                        f"op={metadata.get('operation_type')} lag_ms={capture_lag_ms} "
                        f"threshold_ms={warn_threshold}",
                    )

        runtime["cdc_seen"] = int(runtime.get("cdc_seen", 0) or 0) + 1

        # Append to pending buffer.
        if isinstance(full_doc, dict) and doc_id_str:
            slim = _slim_doc_for_cache(
                full_doc, doc_id_str, fallback_lm=metadata.get("event_time")
            )
            async with runtime["buffer_lock"]:
                cid_buf = runtime["pending_buffer"].setdefault(conversation_id, {})
                # If same _id already pending, keep freshest by _lm.
                existing_in_buf = cid_buf.get(doc_id_str)
                if existing_in_buf is not None:
                    existing_lm = _parse_iso_safe(existing_in_buf.get("_lm"))
                    incoming_lm = _parse_iso_safe(slim.get("_lm"))
                    if existing_lm and incoming_lm and incoming_lm < existing_lm:
                        runtime["dirty_cids"].add(conversation_id)
                        runtime["wakeup_event"].set()
                        return
                cid_buf[doc_id_str] = slim
                runtime["dirty_cids"].add(conversation_id)
            runtime["wakeup_event"].set()
        else:
            # No fullDocument — still mark dirty so consumer rebuilds from cache.
            async with runtime["buffer_lock"]:
                runtime["pending_buffer"].setdefault(conversation_id, {})
                runtime["dirty_cids"].add(conversation_id)
            runtime["wakeup_event"].set()

    # ------------------------------------------------------------------
    # CDC CONSUMER — coalescing loop
    # ------------------------------------------------------------------

    @classmethod
    async def _drain_pending_batch(
        cls, runtime: dict
    ) -> tuple[set[str], dict[str, dict[str, dict]]]:
        """Drain whatever is currently in the pending buffer.

        - If something dirty: snapshot it and return immediately.
        - If nothing: short bounded wait (max_idle_wait), then return what's
          there (possibly still empty — caller just loops).

        Coalescing core: we never sleep when there's work. Natural processing
        time of the previous iteration IS the accumulation window for this one.
        """
        async with runtime["buffer_lock"]:
            available = runtime["dirty_cids"] - runtime["in_flight_cids"]
            if available:
                return cls._snapshot_chunk(runtime, available)
            runtime["wakeup_event"].clear()

        try:
            await asyncio.wait_for(
                runtime["wakeup_event"].wait(),
                timeout=runtime["max_idle_wait"],
            )
        except asyncio.TimeoutError:
            pass

        async with runtime["buffer_lock"]:
            available = runtime["dirty_cids"] - runtime["in_flight_cids"]
            if not available:
                return set(), {}
            return cls._snapshot_chunk(runtime, available)

    @staticmethod
    def _snapshot_chunk(
        runtime: dict, available: set[str]
    ) -> tuple[set[str], dict[str, dict[str, dict]]]:
        limit = runtime["max_batch_cids"]
        if len(available) > limit:
            taken = set(list(available)[:limit])
        else:
            taken = set(available)

        docs_per_cid: dict[str, dict[str, dict]] = {}
        for cid in taken:
            docs_per_cid[cid] = runtime["pending_buffer"].pop(cid, {})
            runtime["dirty_cids"].discard(cid)
            runtime["in_flight_cids"].add(cid)
        return taken, docs_per_cid

    @classmethod
    async def _process_cdc_batch(
        cls,
        runtime: dict,
        cids: set[str],
        pending_docs: dict[str, dict[str, dict]],
    ) -> None:
        if not cids:
            return

        logger = runtime.get("logger")
        redis_client = runtime["redis_client"]
        redis_prefix = runtime["redis_prefix"]
        cid_list = list(cids)

        try:
            await wait_until_processing_unpaused(runtime["request_proxy"].app)
            await begin_processing_activity(runtime["request_proxy"].app)
            try:
                # === 1. Pipelined read =================================
                doc_caches, existing_by_cid = await redis_client.cdc_batch_read(
                    cid_list, key_prefix=redis_prefix
                )

                # === 2. Merge cached + pending with OOO fence ==========
                merged_docs_by_cid: dict[str, list[dict]] = {}
                doc_cache_upserts: dict[str, dict[str, dict]] = {}

                for cid in cid_list:
                    cached = doc_caches.get(cid) or {}
                    pending = pending_docs.get(cid) or {}

                    final_docs: dict[str, dict] = dict(cached)
                    upserts: dict[str, dict] = {}

                    for doc_id, new_doc in pending.items():
                        existing_doc = final_docs.get(doc_id)
                        if existing_doc is not None:
                            existing_lm = _parse_iso_safe(existing_doc.get("_lm"))
                            incoming_lm = _parse_iso_safe(new_doc.get("_lm"))
                            if existing_lm and incoming_lm and incoming_lm < existing_lm:
                                continue  # stale — drop
                        final_docs[doc_id] = new_doc
                        upserts[doc_id] = new_doc

                    if upserts:
                        doc_cache_upserts[cid] = upserts

                    aggregator_input = [
                        {k: v for k, v in doc.items() if k != "_lm"}
                        for doc in final_docs.values()
                    ]
                    merged_docs_by_cid[cid] = aggregator_input

                # === 3. Concurrent aggregation =========================
                aggregated_writes: dict[str, dict] = {}
                deletes: list[str] = []
                failed_cids: set[str] = set()

                async def _aggregate_one(cid: str) -> None:
                    async with runtime["agg_sem"]:
                        try:
                            docs = merged_docs_by_cid.get(cid) or []
                            if not docs:
                                if existing_by_cid.get(cid) is not None:
                                    deletes.append(cid)
                                return
                            aggregated = runtime["metrics"]._build_aggregated_payload(docs)
                            if not aggregated:
                                return
                            merged = runtime["metrics"]._merge_with_existing(
                                existing_by_cid.get(cid), aggregated
                            )
                            aggregated_writes[cid] = merged
                        except Exception as exc:
                            failed_cids.add(cid)
                            runtime["cdc_processing_failures_total"] = int(
                                runtime.get("cdc_processing_failures_total", 0) or 0
                            ) + 1
                            _increment_reason_counter(
                                runtime, "cdc_processing_failures_by_reason", "aggregation_error"
                            )
                            await _safe_async_log(
                                logger.error if logger else None,
                                f"CDC aggregation failed for cid={cid}: {exc}",
                            )

                await asyncio.gather(*[_aggregate_one(cid) for cid in cid_list])

                # === 4. Detect closed conversations ====================
                closed_cids: set[str] = set()
                for cid, payload in aggregated_writes.items():
                    if _is_payload_closed(payload):
                        closed_cids.add(cid)

                # === 5. Pipelined write ================================
                if aggregated_writes or doc_cache_upserts or closed_cids or deletes:
                    aggregated_for_write = {
                        cid: payload for cid, payload in aggregated_writes.items()
                        if cid not in deletes
                    }
                    await redis_client.cdc_batch_write(
                        doc_cache_upserts=doc_cache_upserts,
                        aggregated_writes=aggregated_for_write,
                        closed_conversations=closed_cids,
                        key_prefix=redis_prefix,
                    )
                    if deletes:
                        await redis_client.delete_interactions_batch(
                            deletes, key_prefix=redis_prefix
                        )
                        for cid in deletes:
                            try:
                                await redis_client.delete_doc_cache(cid, key_prefix=redis_prefix)
                            except Exception:
                                pass

                # === 6. Enqueue totals delta ===========================
                if aggregated_writes or deletes:
                    deleted_states = {
                        cid: existing_by_cid.get(cid)
                        for cid in deletes
                        if existing_by_cid.get(cid) is not None
                    }
                    await runtime["totals_queue"].put(
                        _normalize_totals_delta_event(
                            previous_states=existing_by_cid,
                            current_states=aggregated_writes,
                            deleted_states=deleted_states,
                        )
                    )
                    runtime["cdc_totals_enqueue_total"] = int(
                        runtime.get("cdc_totals_enqueue_total", 0) or 0
                    ) + 1

                runtime["cdc_batches_processed"] = int(
                    runtime.get("cdc_batches_processed", 0) or 0
                ) + 1
                _update_app_cdc_status(runtime)

                # Re-queue failed cids for retry.
                if failed_cids:
                    async with runtime["buffer_lock"]:
                        for cid in failed_cids:
                            if cid in pending_docs:
                                cid_buf = runtime["pending_buffer"].setdefault(cid, {})
                                cid_buf.update(pending_docs[cid])
                            runtime["dirty_cids"].add(cid)
                    runtime["wakeup_event"].set()

            finally:
                await end_processing_activity(runtime["request_proxy"].app)
        finally:
            async with runtime["buffer_lock"]:
                for cid in cids:
                    runtime["in_flight_cids"].discard(cid)

    @classmethod
    async def _run_cdc_consumer(cls, runtime: dict) -> None:
        """Single coalescing consumer — no fixed sleep, immediate looping."""
        while True:
            try:
                cids, pending_docs = await cls._drain_pending_batch(runtime)
                if not cids:
                    continue
                await cls._process_cdc_batch(runtime, cids, pending_docs)
            except asyncio.CancelledError:
                raise
            except Exception as exc:
                logger = runtime.get("logger")
                await _safe_async_log(
                    logger.error if logger else None,
                    f"CDC consumer iteration failed: {exc}",
                )
                await asyncio.sleep(0.5)

    # ------------------------------------------------------------------
    # CDC TOTALS CONSUMER (unchanged shape)
    # ------------------------------------------------------------------

    @staticmethod
    async def _collect_totals_batch(queue: asyncio.Queue, batch_window: float) -> list[dict]:
        event = await queue.get()
        batch = [event]
        queue.task_done()
        start = time_module.monotonic()
        while True:
            remaining = batch_window - (time_module.monotonic() - start)
            if remaining <= 0:
                return batch
            try:
                nxt = await asyncio.wait_for(queue.get(), timeout=remaining)
                batch.append(nxt)
                queue.task_done()
            except asyncio.TimeoutError:
                return batch

    @classmethod
    async def _run_totals_batch_consumer(cls, runtime: dict) -> None:
        app_ref = runtime["request_proxy"].app
        while True:
            try:
                batch = await cls._collect_totals_batch(
                    runtime["totals_queue"], runtime["totals_batch_window"]
                )
                previous_states, current_states, deleted_states = _merge_totals_delta_events(batch)
                if not current_states and not deleted_states:
                    continue
                await wait_until_processing_unpaused(app_ref)
                await begin_processing_activity(app_ref)
                try:
                    await runtime["metrics"]._update_total_metrics_batch(
                        previous_states,
                        current_states,
                        deleted_states=deleted_states or None,
                    )
                    runtime["cdc_totals_batches_processed"] = int(
                        runtime.get("cdc_totals_batches_processed", 0) or 0
                    ) + 1
                    _update_app_cdc_status(runtime)
                finally:
                    await end_processing_activity(app_ref)
            except asyncio.CancelledError:
                raise
            except Exception as exc:
                logger = runtime.get("logger")
                await _safe_async_log(
                    logger.error if logger else None,
                    f"Totals consumer failed: {exc}",
                )
                await asyncio.sleep(0.5)

    # CDC stream + supervisor
    # ------------------------------------------------------------------

    @classmethod
    async def _watch_cdc_stream(cls, runtime: dict) -> None:
        pipeline = [{"$match": {"operationType": {"$in": ["insert", "update", "replace"]}}}]
        logger = runtime.get("logger")
        watch_kwargs: dict[str, object] = {}
        if runtime.get("resume_token") is not None:
            watch_kwargs["resume_after"] = runtime.get("resume_token")
        watch_kwargs["max_await_time_ms"] = max(
            1000, int(os.getenv("CDC_STREAM_MAX_AWAIT_MS", "5000"))
        )
        # CRITICAL: ensures fullDocument is populated on update events.
        watch_kwargs["full_document"] = "updateLookup"

        await _safe_async_log(
            logger.info if logger else None,
            f"CDC stream: watching {runtime['target_db']}.{runtime['collectionname']} "
            f"(full_document=updateLookup, idle_wait_ms={runtime['max_idle_wait_ms']}, "
            f"max_batch_cids={runtime['max_batch_cids']}, "
            f"resuming={runtime.get('resume_token') is not None})",
        )

        async with runtime["collection"].watch(pipeline, **watch_kwargs) as stream:
            async for change in stream:
                runtime["cdc_last_event_received_at"] = datetime.now(timezone.utc)
                await cls._handle_cdc_change(runtime, change)
                if isinstance(change, dict):
                    runtime["resume_token"] = change.get("_id") or runtime.get("resume_token")
                else:
                    runtime["resume_token"] = (
                        getattr(stream, "resume_token", None) or runtime.get("resume_token")
                    )
                runtime["cdc_last_resume_token_seen"] = (
                    str(runtime["resume_token"])
                    if runtime["resume_token"] is not None else None
                )

    @classmethod
    async def _start_cdc_consumers(cls, runtime: dict) -> list[asyncio.Task]:
        consumer_tasks = [
            asyncio.create_task(cls._run_cdc_consumer(runtime)),
            asyncio.create_task(cls._run_totals_batch_consumer(runtime)),
        ]
        logger = runtime.get("logger")
        await _safe_async_log(
            logger.info if logger else None,
            f"CDC consumers started for {runtime['target_db']}.{runtime['collectionname']} "
            f"(idle_wait_ms={runtime['max_idle_wait_ms']}, "
            f"max_batch_cids={runtime['max_batch_cids']}, "
            f"totals_window_ms={runtime['totals_batch_window_ms']})",
        )
        await _safe_async_log(
            logger.info if logger else None,
            f"CDC filter criteria for prefix {runtime['redis_prefix']}: {runtime['cdc_filter_criteria']}",
        )
        _update_app_cdc_status(runtime)
        return consumer_tasks

    @staticmethod
    async def _cancel_cdc_consumers(consumer_tasks: list[asyncio.Task]) -> None:
        for task in consumer_tasks:
            task.cancel()
        if consumer_tasks:
            await asyncio.gather(*consumer_tasks, return_exceptions=True)

    @staticmethod
    def _snapshot_cdc_resume_token(runtime: dict | None):
        if not runtime:
            return None
        return runtime.get("resume_token")

    @staticmethod
    def _snapshot_cdc_carryover(
        runtime: dict | None,
    ) -> tuple[set[str], dict[str, dict[str, dict]]]:
        if not runtime:
            return set(), {}
        cids = set(runtime.get("dirty_cids") or set()) | set(runtime.get("in_flight_cids") or set())
        pending = {
            cid: dict(runtime.get("pending_buffer", {}).get(cid, {}))
            for cid in cids
        }
        return cids, pending

    @classmethod
    async def clear_cdc_runtime_state(cls, runtime: dict | None) -> dict[str, int]:
        if not isinstance(runtime, dict):
            return {"pending": 0, "dirty": 0, "in_flight": 0}
        async with runtime["buffer_lock"]:
            pending_count = sum(
                len(v) for v in (runtime.get("pending_buffer") or {}).values()
            )
            dirty_count = len(runtime.get("dirty_cids") or set())
            in_flight_count = len(runtime.get("in_flight_cids") or set())
            if isinstance(runtime.get("pending_buffer"), dict):
                runtime["pending_buffer"].clear()
            if isinstance(runtime.get("dirty_cids"), set):
                runtime["dirty_cids"].clear()
            if isinstance(runtime.get("in_flight_cids"), set):
                runtime["in_flight_cids"].clear()
        return {"pending": pending_count, "dirty": dirty_count, "in_flight": in_flight_count}

    @classmethod
    async def start_cdc_stream(
        cls,
        app,
        *,
        dbname: str | None = None,
        collectionname: str = "anirudh_conversation_test2",
        redis_prefix: str = DEFAULT_CONVERSATION_REDIS_PREFIX,
        max_concurrency: int = 40,
        retry_delay_seconds: int = 5,
    ):
        logger = getattr(app.state, "logger", None)
        resume_token = None
        restart_count = 0
        restart_reason: str | None = None
        carryover_cids: set[str] = set()
        carryover_pending: dict[str, dict[str, dict]] = {}

        while True:
            runtime = None
            consumer_tasks: list[asyncio.Task] = []
            try:
                runtime = await cls._build_cdc_runtime(
                    app,
                    dbname,
                    collectionname,
                    redis_prefix,
                    max_concurrency,
                    resume_token=resume_token,
                    restart_count=restart_count,
                    restart_reason=restart_reason,
                )
                app.state.cdc_runtime = runtime

                if carryover_cids:
                    async with runtime["buffer_lock"]:
                        for cid in carryover_cids:
                            buf = runtime["pending_buffer"].setdefault(cid, {})
                            buf.update(carryover_pending.get(cid, {}))
                            runtime["dirty_cids"].add(cid)
                    runtime["wakeup_event"].set()
                    await _safe_async_log(
                        logger.warning if logger else None,
                        f"CDC stream restarting; queued {len(carryover_cids)} cid(s) for replay",
                    )
                    carryover_cids = set()
                    carryover_pending = {}

                consumer_tasks = await cls._start_cdc_consumers(runtime)
                await cls._watch_cdc_stream(runtime)
            except Exception as exc:
                restart_count += 1
                restart_reason = str(exc)
                await _safe_async_log(
                    logger.error if logger else None,
                    f"CDC stream error for {collectionname}: {exc}",
                )
                if resume_token is not None and _is_invalid_resume_token_error(exc):
                    await _safe_async_log(
                        logger.warning if logger else None,
                        "CDC resume token invalid; restarting without resume token",
                    )
                    resume_token = None
                await asyncio.sleep(max(1, retry_delay_seconds))
            finally:
                if getattr(app.state, "cdc_runtime", None) is runtime:
                    app.state.cdc_runtime = None
                carryover_cids, carryover_pending = cls._snapshot_cdc_carryover(runtime)
                resume_token = cls._snapshot_cdc_resume_token(runtime) or resume_token
                await cls._cancel_cdc_consumers(consumer_tasks)

    # ------------------------------------------------------------------
    # process_conversation_ids_batch — coverage reconciliation path
    # ------------------------------------------------------------------

    @classmethod
    async def process_conversation_ids_batch(
        cls,
        app,
        *,
        conversation_ids: list[str],
        dbname: str | None = None,
        collectionname: str = "conversation",
        redis_prefix: str = DEFAULT_CONVERSATION_REDIS_PREFIX,
        max_concurrency: int = 40,
    ) -> int:
        unique_ids = [cid for cid in dict.fromkeys(conversation_ids) if cid]
        if not unique_ids:
            return 0

        from app.core.redis_connect import AWSRedis
        from app.service.realtime_metrics import RealtimeMetrics

        target_db = cls._resolve_cdc_target_db(dbname)
        request_proxy = _RequestProxy(app)
        redis_client = AWSRedis()
        metrics = RealtimeMetrics(
            redis_client=redis_client,
            logger=getattr(app.state, "logger", None),
            redis_key_prefix=redis_prefix,
        )

        docs_by_cid = await cls.get_documents_grouped_by_conversation_ids(
            request_proxy, target_db, collectionname, unique_ids
        )

        doc_cache_updates: dict[str, dict[str, dict]] = {}
        aggregated_writes: dict[str, dict] = {}
        existing_by_cid = await redis_client.get_interactions_batch(
            unique_ids, key_prefix=redis_prefix
        )
        closed_cids: set[str] = set()

        for cid in unique_ids:
            docs = docs_by_cid.get(cid) or []
            if not docs:
                continue
            doc_map: dict[str, dict] = {}
            for doc in docs:
                doc_id = str(doc.get("_id") or doc.get("conversationId"))
                slim: dict = {}
                for field in _CDC_CACHED_FIELDS:
                    if field == "_id":
                        slim["_id"] = doc_id
                        continue
                    if field in doc:
                        slim[field] = doc[field]
                slim["_lm"] = doc.get("lastModifiedDateTime") or datetime.now(timezone.utc).isoformat()
                doc_map[doc_id] = slim
            doc_cache_updates[cid] = doc_map

            aggregator_input = [{k: v for k, v in d.items() if k != "_lm"} for d in doc_map.values()]
            aggregated = metrics._build_aggregated_payload(aggregator_input)
            if not aggregated:
                continue
            merged = metrics._merge_with_existing(existing_by_cid.get(cid), aggregated)
            aggregated_writes[cid] = merged
            if _is_payload_closed(merged):
                closed_cids.add(cid)

        if doc_cache_updates or aggregated_writes:
            await redis_client.cdc_batch_write(
                doc_cache_upserts=doc_cache_updates,
                aggregated_writes=aggregated_writes,
                closed_conversations=closed_cids,
                key_prefix=redis_prefix,
            )

        if aggregated_writes:
            await metrics._update_total_metrics_batch(
                existing_by_cid,
                aggregated_writes,
            )

        return len(unique_ids)

    # ------------------------------------------------------------------
    # has_recent_source_activity (unchanged)
    # ------------------------------------------------------------------

    @classmethod
    async def has_recent_source_activity(
        cls,
        dbname: str | None,
        collectionname: str,
        *,
        lookback_seconds: int,
    ) -> bool:
        target_db = cls._resolve_cdc_target_db(dbname)
        collection = await cls.get_database_collection(target_db, collectionname, client_role="query")
        threshold = datetime.now(timezone.utc) - timedelta(seconds=max(1, int(lookback_seconds)))
        threshold_iso = threshold.isoformat()
        query = with_allowed_customer_id_filter(
            {
                "$or": [
                    {"lastModifiedDateTime": {"$gte": threshold}},
                    {"lastModifiedDateTime": {"$gte": threshold_iso}},
                    {"createdDateTime": {"$gte": threshold}},
                    {"createdDateTime": {"$gte": threshold_iso}},
                ]
            }
        )
        return bool(await collection.find_one(query, {"_id": 1}))

    # ------------------------------------------------------------------
    # Redis dump (skips :doccache: keys — internal scratch space)
    # ------------------------------------------------------------------

    @classmethod
    async def _build_redis_dump_runtime(cls, dbname: str | None, target_collection: str):
        from app.core.redis_connect import AWSRedis
        redis_client = AWSRedis()
        redis_scan_count = int(os.getenv("REDIS_DUMP_SCAN_COUNT", "1000"))
        target_db = cls._resolve_cdc_target_db(dbname)
        collection = await cls.get_database_collection(target_db, target_collection)
        return redis_client, redis_scan_count, target_db, collection

    @staticmethod
    def _build_redis_dump_document(full_key: str, payload, dumped_at: datetime) -> dict:
        doc = payload if isinstance(payload, dict) else {"value": payload}
        doc = dict(doc)
        doc["_redis_key"] = full_key
        doc["_dumped_at_utc"] = dumped_at
        return doc

    @classmethod
    async def _flush_redis_dump_batch(
        cls, logger, collection, redis_client,
        pending_docs, pending_keys, delete_after_dump,
    ):
        if not pending_docs:
            return 0
        docs_to_insert = list(pending_docs)
        keys_to_delete = list(pending_keys)
        pending_docs.clear()
        pending_keys.clear()
        try:
            res = await collection.insert_many(docs_to_insert, ordered=False)
            written = len(res.inserted_ids)
        except Exception as batch_exc:
            if logger:
                await logger.error(f"Redis dump batch insert failed: {batch_exc}")
            return 0
        if delete_after_dump and keys_to_delete:
            try:
                await redis_client.delete_keys(keys_to_delete)
            except Exception as del_exc:
                if logger:
                    await logger.error(f"Redis dump cleanup failed: {del_exc}")
        return written

    @classmethod
    async def _scan_redis_prefix_for_dump(
        cls, logger, redis_client, collection, redis_prefix, redis_scan_count,
        ttl_extension_seconds, delete_after_dump, batch_size,
    ):
        inserted = 0
        scanned = 0
        pending_docs: list[dict] = []
        pending_keys: list[str] = []
        async for items in redis_client.iter_prefix_items(
            redis_prefix, scan_count=redis_scan_count,
            extend_ttl_seconds=ttl_extension_seconds,
        ):
            dumped_at = datetime.now(timezone.utc)
            for full_key, payload in items.items():
                if ":doccache:" in full_key:
                    continue
                scanned += 1
                pending_docs.append(cls._build_redis_dump_document(full_key, payload, dumped_at))
                pending_keys.append(full_key)
                if len(pending_docs) < batch_size:
                    continue
                inserted += await cls._flush_redis_dump_batch(
                    logger, collection, redis_client,
                    pending_docs, pending_keys, delete_after_dump,
                )
        inserted += await cls._flush_redis_dump_batch(
            logger, collection, redis_client,
            pending_docs, pending_keys, delete_after_dump,
        )
        return inserted, scanned

    @classmethod
    async def dump_redis_prefix_to_mongo(
        cls, app, *,
        dbname: str | None = None,
        redis_prefix: str = DEFAULT_CONVERSATION_REDIS_PREFIX,
        target_collection: str = "conversation_anirudh_download2",
        ttl_extension_seconds: int = 900,
        delete_after_dump: bool = True,
        batch_size: int = 1000,
    ):
        logger = getattr(app.state, "logger", None)
        try:
            await wait_until_processing_unpaused(app)
            await begin_processing_activity(app)
            redis_client, redis_scan_count, target_db, collection = await cls._build_redis_dump_runtime(
                dbname, target_collection,
            )
            inserted, scanned = await cls._scan_redis_prefix_for_dump(
                logger, redis_client, collection, redis_prefix, redis_scan_count,
                ttl_extension_seconds, delete_after_dump, batch_size,
            )
        except Exception as exc:
            if logger:
                await logger.error(f"Redis dump failed for prefix {redis_prefix}: {exc}")
            return 0
        finally:
            await end_processing_activity(app)
        if scanned == 0:
            if logger:
                await logger.info(f"Redis dump: no keys found for prefix {redis_prefix}")
            return 0
        if logger:
            await logger.info(
                f"Redis dump complete: {inserted} docs from prefix {redis_prefix} "
                f"into {target_db}.{target_collection} (scanned={scanned})"
            )
        return inserted

    @classmethod
    async def schedule_daily_redis_dump(
        cls, app, *,
        redis_prefix: str = DEFAULT_CONVERSATION_REDIS_PREFIX,
        target_collection: str = "conversation_anirudh_download2",
        lead_seconds: int = 30,
        ttl_extension_seconds: int = 900,
    ):
        logger = getattr(app.state, "logger", None)

        def seconds_until_next_midnight_utc():
            now = datetime.now(timezone.utc)
            next_midnight = (now + timedelta(days=1)).replace(
                hour=0, minute=0, second=0, microsecond=0
            )
            return max(1, int((next_midnight - now).total_seconds()))

        while True:
            try:
                sleep_for = max(1, seconds_until_next_midnight_utc() - max(0, lead_seconds))
                if logger:
                    await logger.info(
                        f"Redis dump scheduler sleeping {sleep_for}s before next run "
                        f"(prefix={redis_prefix}, collection={target_collection})"
                    )
                await asyncio.sleep(sleep_for)
                if logger:
                    await logger.info(
                        f"Starting scheduled Redis dump "
                        f"(prefix={redis_prefix}, collection={target_collection})"
                    )
                await cls.dump_redis_prefix_to_mongo(
                    app, redis_prefix=redis_prefix, target_collection=target_collection,
                    ttl_extension_seconds=ttl_extension_seconds, delete_after_dump=True,
                )
            except Exception as exc:
                if logger:
                    await logger.error(f"Scheduled Redis dump failed: {exc}")
            finally:
                await asyncio.sleep(5)
