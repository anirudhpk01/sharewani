# ---------------------------------------------------------------------------
# Why this file exists
# ---------------------------------------------------------------------------
# All Redis I/O in this project goes through this wrapper.
# Keeping connection details, key-prefix logic, serialization, and TTL rules
# in one place means the rest of the app never has to think about them.
#
# Key design decisions explained inline:
#
# SINGLETON POOL
#   Each AWSRedis() used to build a new Redis client with its own connection
#   pool.  Under burst CDC load that meant dozens of pools opening at once,
#   exhausting sockets and triggering TimeoutError.  We now share ONE pool
#   process-wide.  AWSRedis() is still cheap to construct; it just reuses the
#   shared pool.
#
# CHUNKED MGET / PIPELINE
#   A single mget(*14500_keys) or a pipeline of 14500 SET commands can easily
#   exceed the socket_timeout.  We split every bulk operation into chunks of
#   REDIS_MGET_CHUNK_SIZE keys (default 200).  Each chunk is a small, fast
#   round-trip.  The caller sees no difference — it still gets one dict back.
#
# RAISED TIMEOUTS
#   socket_timeout=3 was too tight for a loaded Redis under pipeline pressure.
#   We raise defaults to 10s connect / 15s socket.  Both are still env-var
#   overridable so production tuning does not require a code change.
# ---------------------------------------------------------------------------

# ---------------------------------------------------------------------------
# Singleton pool — one per process
# ---------------------------------------------------------------------------
_POOL_LOCK = asyncio.Lock()
_SHARED_POOL: ConnectionPool | None = None


def _get_or_create_pool() -> ConnectionPool:
    """Return the process-wide Redis ConnectionPool, creating it on first call.

    This is called from a synchronous context (inside __init__) so we cannot
    await here.  Instead we use a module-level reference and a best-effort
    check.  The asyncio.Lock guard in get_shared_pool() handles the true
    concurrency case; this sync version is fine for the common path where the
    pool was already created.
    """
    global _SHARED_POOL
    if _SHARED_POOL is not None:
        return _SHARED_POOL

    config = ConfigReader().get_redis_config()
    host = config.get("host")
    port = config.get("port")
    username = config.get("username")
    password = config.get("redis_auth")

    try:
        max_connections = int(config.get("max_connections") or 300)
    except Exception:
        max_connections = 300

    connect_timeout = float(config.get("redis_connect_timeout") or
                            __import__("os").getenv("REDIS_CONNECT_TIMEOUT", "10"))
    socket_timeout = float(config.get("redis_socket_timeout") or
                           __import__("os").getenv("REDIS_SOCKET_TIMEOUT", "15"))

    retry_strategy = Retry(ExponentialBackoff(base=1, cap=10), 3)

    _SHARED_POOL = ConnectionPool(
        host=host,
        port=port,
        username=username,
        password=password,
        decode_responses=True,
        ssl=True,
        ssl_cert_reqs="none",
        max_connections=max_connections,
        retry=retry_strategy,
        retry_on_error=[ConnectionError, TimeoutError],
        socket_connect_timeout=connect_timeout,
        socket_timeout=socket_timeout,
        health_check_interval=30,
    )
    return _SHARED_POOL


# How many Redis keys to include in a single mget or pipeline batch.
# Keeping this small prevents any one Redis round-trip from timing out under load.
import os as _os
_MGET_CHUNK = int(_os.getenv("REDIS_MGET_CHUNK_SIZE", "200"))
_PIPELINE_CHUNK = int(_os.getenv("REDIS_PIPELINE_CHUNK_SIZE", "200"))


class AWSRedis:
    """Async Redis wrapper used across the service.

    Responsibilities:
    - reuse the shared connection pool (no new pool per instance)
    - build fully-qualified Redis keys with the correct prefix
    - serialize Python dicts to JSON on write
    - parse JSON back into Python objects on read
    - apply default TTL rules for conversation keys
    - chunk bulk operations so no single call can time out under burst load
    """

    def __init__(self):
        logger.debug("AWSRedis: attaching to shared connection pool")
        pool = _get_or_create_pool()
        self._client = Redis(connection_pool=pool)

    # ------------------------------------------------------------------
    # Internal helpers
    # ------------------------------------------------------------------

    @staticmethod
    def _seconds_until_next_midnight_utc() -> int:
        """Seconds from now (UTC) until the next 00:00 UTC boundary."""
        now = datetime.now(timezone.utc)
        next_midnight = (now + timedelta(days=1)).replace(
            hour=0, minute=0, second=0, microsecond=0
        )
        return max(1, int((next_midnight - now).total_seconds()))

    def _build_prefix(self, key_prefix: str | None = None) -> str:
        """Return the fully-qualified prefix string, always ending with ':'."""
        config = ConfigReader().get_redis_config()
        base = config.get("key")
        prefix = f"{base}:{key_prefix}" if key_prefix is not None else base
        return str(prefix).rstrip(":") + ":"

    @staticmethod
    def _parse_redis_value(value: Any) -> Any:
        if value is None:
            return None
        try:
            return json.loads(value)
        except Exception:
            return value

    # ------------------------------------------------------------------
    # Single-key operations
    # ------------------------------------------------------------------

    async def write_interaction(
        self,
        key: str,
        value: dict,
        expire_seconds: int | None = None,
        key_prefix: str | None = None,
    ) -> None:
        """Write one conversation key to Redis with automatic midnight TTL."""
        prefix = self._build_prefix(key_prefix)
        full_key = prefix + key
        ttl = expire_seconds if expire_seconds is not None else self._seconds_until_next_midnight_utc()
        try:
            await self._client.set(full_key, json.dumps(value), ex=int(ttl))
        except Exception as exc:
            logger.error(f"Failed to write Redis key {full_key}: {exc}")
            raise  # re-raise so callers can decide whether to retry or skip

    async def get_interaction(self, key: str, key_prefix: str | None = None) -> Any:
        """Read one conversation key from Redis."""
        prefix = self._build_prefix(key_prefix)
        full_key = prefix + key
        try:
            val = await self._client.get(full_key)
        except AuthenticationError:
            val = await self._client.get(full_key)
        return self._parse_redis_value(val)

    async def delete_interaction(self, key: str, key_prefix: str | None = None) -> int:
        prefix = self._build_prefix(key_prefix)
        full_key = prefix + key
        try:
            return await self._client.delete(full_key)
        except AuthenticationError:
            return await self._client.delete(full_key)

    async def delete_keys(self, keys: list[str]) -> int:
        """Delete multiple raw (fully-qualified) Redis keys."""
        if not keys:
            return 0
        try:
            return await self._client.delete(*keys)
        except AuthenticationError:
            return await self._client.delete(*keys)

    # ------------------------------------------------------------------
    # Batch operations — chunked to avoid timeout under burst load
    # ------------------------------------------------------------------

    async def get_interactions_batch(
        self, keys: list[str], key_prefix: str | None = None
    ) -> dict[str, Any]:
        """Read many conversation keys and return {logical_key: parsed_value}.

        The underlying mget is split into chunks of _MGET_CHUNK so that no
        single Redis call carries thousands of keys and risks a timeout.
        """
        if not keys:
            return {}

        prefix = self._build_prefix(key_prefix)
        result: dict[str, Any] = {}

        for start in range(0, len(keys), _MGET_CHUNK):
            chunk_keys = keys[start : start + _MGET_CHUNK]
            full_keys = [prefix + k for k in chunk_keys]
            try:
                values = await self._client.mget(*full_keys)
            except AuthenticationError:
                values = await self._client.mget(*full_keys)

            for key, value in zip(chunk_keys, values):
                result[key] = self._parse_redis_value(value)

        return result

    async def write_interactions_batch(
        self,
        items: dict[str, dict],
        expire_seconds: int | None = None,
        key_prefix: str | None = None,
    ) -> int:
        """Write many conversation keys in chunked pipelines.

        Each pipeline chunk contains at most _PIPELINE_CHUNK SET commands.
        This prevents any single pipeline execution from timing out when
        hundreds of conversations are written at once during CDC burst load.
        """
        if not items:
            return 0

        prefix = self._build_prefix(key_prefix)
        ttl = expire_seconds if expire_seconds is not None else self._seconds_until_next_midnight_utc()
        item_list = list(items.items())
        written = 0

        for start in range(0, len(item_list), _PIPELINE_CHUNK):
            chunk = item_list[start : start + _PIPELINE_CHUNK]
            pipe = self._client.pipeline(transaction=False)
            for key, value in chunk:
                pipe.set(prefix + key, json.dumps(value), ex=int(ttl))
            try:
                await pipe.execute()
                written += len(chunk)
            except AuthenticationError:
                # Rebuild and retry once for auth-specific transient failures.
                pipe = self._client.pipeline(transaction=False)
                for key, value in chunk:
                    pipe.set(prefix + key, json.dumps(value), ex=int(ttl))
                await pipe.execute()
                written += len(chunk)
            # A brief yield between pipeline chunks lets other coroutines (e.g. the
            # CDC stream reader) run and prevents the event loop from stalling.
            await asyncio.sleep(0)

        return written

    # ------------------------------------------------------------------
    # Wide scan — used by get_all_with_prefix and Redis dump
    # ------------------------------------------------------------------

    async def get_all_with_prefix(
        self,
        prefix: str,
        *,
        extend_ttl_seconds: int | None = None,
        delete_after: bool = False,
    ) -> dict:
        """Return {full_redis_key: parsed_value} for all keys matching prefix.

        Uses SCAN (non-blocking) to collect keys, then fetches values in
        chunked mget calls so no single call carries thousands of keys.

        Optional extend_ttl_seconds refreshes TTL before reading (useful near
        midnight).  Optional delete_after removes keys after fetching (for
        post-dump cleanup).
        """
        p = self._build_prefix(prefix)
        pattern = p + "*"

        try:
            # SCAN iterates the keyspace incrementally — never blocks Redis.
            keys: list[str] = []
            async for k in self._client.scan_iter(match=pattern, count=500):
                keys.append(k)

            if not keys:
                return {}

            # Optionally refresh TTLs in chunked pipelines.
            if extend_ttl_seconds and extend_ttl_seconds > 0:
                for start in range(0, len(keys), _PIPELINE_CHUNK):
                    chunk = keys[start : start + _PIPELINE_CHUNK]
                    pipe = self._client.pipeline(transaction=False)
                    for k in chunk:
                        pipe.expire(k, int(extend_ttl_seconds))
                    await pipe.execute()
                    await asyncio.sleep(0)

            # Fetch values in chunks — this is the fix for the GET /conversations
            # timeout.  Previously a single mget(*14500_keys) would time out.
            result: dict[str, Any] = {}
            for start in range(0, len(keys), _MGET_CHUNK):
                chunk = keys[start : start + _MGET_CHUNK]
                vals = await self._client.mget(*chunk)
                for k, v in zip(chunk, vals):
                    try:
                        parsed = json.loads(v) if v is not None else None
                    except Exception:
                        parsed = v
                    result[k] = parsed
                # Yield between chunks to keep the event loop responsive.
                await asyncio.sleep(0)

            if delete_after:
                # Delete in chunks too — same safety reason as above.
                for start in range(0, len(keys), _PIPELINE_CHUNK):
                    chunk = keys[start : start + _PIPELINE_CHUNK]
                    try:
                        await self._client.delete(*chunk)
                    except AuthenticationError:
                        await self._client.delete(*chunk)

            return result

        except Exception as exc:
            logger.error(f"Failed to fetch keys with prefix {p}: {exc}")
            return {}

    # ------------------------------------------------------------------
    # Misc atomic helpers
    # ------------------------------------------------------------------

    async def expire(self, key: str, seconds: int) -> bool:
        return await self._client.expire(key, seconds)

    async def incr(self, key: str) -> int:
        return await self._client.incr(key)

    async def decr(self, key: str) -> int:
        return await self._client.decr(key)

# ---------------------------------------------------------------------------
# Why this file exists
# ---------------------------------------------------------------------------
# This file owns all Mongo-side infrastructure for the service:
#
# 1. One shared Mongo client for the whole process (singleton pattern).
# 2. Document reading helpers used by API and batch flows.
# 3. The CDC loop that keeps Redis updated as Mongo changes arrive.
# 4. The nightly Redis-to-Mongo snapshot dump.
#
# A simple mental model:
#   realtime_metrics.py  — decides WHAT a conversation means
#   connectionpool.py    — decides HOW data moves between Mongo, Redis, and the app
#
# CDC BURST-LOAD FIX (the main change from the previous version)
# ---------------------------------------------------------------
# When 3000-4000 documents arrive at once, the change-stream reader can
# queue hundreds of unique conversationIds very quickly.  The old design fed
# the entire accumulated set to process_batch() at once, which then fired a
# single $in query for all those IDs and a single write_interactions_batch()
# call for all the results.  Under those conditions the Redis pipeline easily
# exceeded the socket timeout.
#
# The fix is a hard cap: CDC_BATCH_PROCESS_SIZE (default 50).  The batch
# consumer slices the accumulated set into sub-batches of at most that size
# and processes them sequentially.  Each sub-batch is a small, fast Mongo
# query and a small, fast Redis pipeline.  The total work is identical; it is
# just spread over many small operations rather than one huge one.
# ---------------------------------------------------------------------------

INTERNAL_SERVER_ERR = "Internal Server Error"
INTERNAL_SERVER_MSG = "An unexpected error occurred. Please try again later."


def get_logger_from_request(request: Request):
    return request.app.state.logger


def _parse_date_value(raw_value: str) -> datetime | None:
    """Parse a date/datetime string into a timezone-aware UTC datetime."""
    if not raw_value:
        return None
    value = raw_value.strip()
    if not value:
        return None
    for fmt in ("%Y-%m-%d", "%d/%m/%Y", "%m/%d/%Y"):
        try:
            dt = datetime.strptime(value, fmt)
            return dt.replace(tzinfo=timezone.utc)
        except Exception:
            pass
    try:
        iso_value = value[:-1] + "+00:00" if value.endswith("Z") else value
        dt = datetime.fromisoformat(iso_value)
        return dt if dt.tzinfo else dt.replace(tzinfo=timezone.utc)
    except Exception:
        return None


def _build_datetime_filter(key_name: str, raw_value) -> dict | None:
    """Build a Mongo range filter for datetime fields.

    Returns None when the field name does not look like a datetime field or
    when parsing fails, so callers can fall back to an equality filter.
    """
    if not key_name or "date" not in key_name.lower():
        return None

    parsed: datetime | None = None
    date_only = False

    if isinstance(raw_value, datetime):
        parsed = raw_value if raw_value.tzinfo else raw_value.replace(tzinfo=timezone.utc)
        date_only = parsed.timetz() == time(0, 0)
    elif isinstance(raw_value, str):
        parsed = _parse_date_value(raw_value)
        date_only = "T" not in raw_value and len(raw_value.strip()) <= 10

    if not parsed:
        return None

    start = parsed.replace(hour=0, minute=0, second=0, microsecond=0) if date_only else parsed
    now_utc = datetime.now(timezone.utc)
    return {"$gte": start, "$lte": now_utc}


class MongoDBConnectionPool:
    """Async singleton Mongo client.

    Call ``await MongoDBConnectionPool.init_pool()`` once at startup.
    Every subsequent call to get_client() / get_database_collection() reuses
    the same underlying AsyncIOMotorClient.
    """

    _client = None
    _config = None
    _lock = asyncio.Lock()

    # ------------------------------------------------------------------
    # Pool lifecycle
    # ------------------------------------------------------------------

    @classmethod
    async def init_pool(cls, config=None):
        async with cls._lock:
            if cls._client is None:
                if config is None:
                    config = ConfigReader().get_mongo_config()
                cls._config = config
                cls._client = AsyncIOMotorClient(
                    config["uri"],
                    maxPoolSize=config["max_pool_size"],
                    minPoolSize=config["min_pool_size"],
                    maxIdleTimeMS=config["idle_timeout_ms"],
                    appname="prompt_manager",
                )

    @classmethod
    async def close_pool(cls):
        if cls._client is not None:
            cls._client.close()
            cls._client = None

    @classmethod
    async def get_client(cls):
        if cls._client is None:
            raise RuntimeError(
                "MongoDBConnectionPool not initialized. Call init_pool() at startup."
            )
        return cls._client

    @classmethod
    async def get_database_collection(cls, dbname, collectionname):
        client = await cls.get_client()
        return client[dbname][collectionname]

    # ------------------------------------------------------------------
    # Serialization helper
    # ------------------------------------------------------------------

    @staticmethod
    def serialize(document):
        """Recursively convert datetime values to ISO strings and strip _id."""
        if isinstance(document, list):
            return [MongoDBConnectionPool.serialize(doc) for doc in document]
        elif isinstance(document, dict):
            doc = dict(document)
            doc.pop("_id", None)
            for key, value in doc.items():
                if isinstance(value, datetime):
                    doc[key] = value.isoformat()
                elif isinstance(value, dict):
                    doc[key] = MongoDBConnectionPool.serialize(value)
                elif isinstance(value, list):
                    doc[key] = MongoDBConnectionPool.serialize(value)
            return doc
        else:
            return document

    # ------------------------------------------------------------------
    # Document read helpers
    # ------------------------------------------------------------------

    @classmethod
    async def get_documents(cls, request: Request, dbname, collectionname, key, key_name):
        """Fetch all documents matching a filter.  Used by API-facing flows."""
        logger = get_logger_from_request(request)
        try:
            collection = await cls.get_database_collection(dbname, collectionname)
            range_filter = _build_datetime_filter(key_name, key)
            if range_filter:
                filter_query = {key_name: range_filter}
                start_dt = range_filter.get("$gte")
                end_dt = range_filter.get("$lte")
                if isinstance(start_dt, datetime) and isinstance(end_dt, datetime):
                    filter_query = {
                        "$or": [
                            {key_name: range_filter},
                            {key_name: {"$gte": start_dt.isoformat(), "$lte": end_dt.isoformat()}},
                        ]
                    }
                await logger.info(
                    f"Applying range filter on {key_name}: "
                    f"{range_filter.get('$gte')} to {range_filter.get('$lte')}"
                )
            else:
                filter_query = {key_name: key}

            projection = {"_id": 0}
            cursor = collection.find(filter_query, projection)
            documents = []
            async for doc in cursor:
                documents.append(doc)
            if documents:
                return cls.serialize(documents)
            return "No data Available"
        except Exception as exc:
            await logger.error(
                f"Failed to retrieve documents from collection '{collectionname}': {exc}"
            )
            return JSONResponse(
                status_code=500,
                content={
                    "error": INTERNAL_SERVER_ERR,
                    "message": INTERNAL_SERVER_MSG,
                    "statusCodeValue": 500,
                },
            )

    @classmethod
    async def get_documents_grouped_by_conversation_ids(
        cls,
        request: Request,
        dbname: str,
        collectionname: str,
        conversation_ids: list[str],
    ) -> dict[str, list[dict]]:
        """Fetch all docs for the given conversationIds in one query, grouped in memory.

        This is the correct helper to call from CDC because it batches multiple
        conversations into a single $in query rather than one query per conversation.
        """
        if not conversation_ids:
            return {}

        logger = get_logger_from_request(request)
        try:
            collection = await cls.get_database_collection(dbname, collectionname)
            projection = {"_id": 0}
            fetch_batch_size = int(os.getenv("CDC_FETCH_CURSOR_BATCH_SIZE", "10000"))
            cursor = collection.find(
                {"conversationId": {"$in": list(conversation_ids)}},
                projection,
            ).sort([("conversationId", 1), ("createdDateTime", 1)])
            cursor = cursor.batch_size(fetch_batch_size)

            # Pre-seed so every requested ID has an entry even if Mongo returns nothing.
            grouped: dict[str, list[dict]] = {cid: [] for cid in conversation_ids}
            async for raw_doc in cursor:
                try:
                    doc = cls.serialize(raw_doc)
                except Exception:
                    doc = raw_doc
                if not isinstance(doc, dict):
                    continue
                cid = doc.get("conversationId")
                if cid:
                    grouped.setdefault(cid, []).append(doc)

            return grouped
        except Exception as exc:
            await logger.error(
                f"Failed to retrieve grouped conversation documents "
                f"from '{collectionname}': {exc}"
            )
            return {}

    # ------------------------------------------------------------------
    # CDC stream
    # ------------------------------------------------------------------

    @classmethod
    async def start_cdc_stream(
        cls,
        app,
        *,
        dbname: str | None = None,
        collectionname: str = "conversation",
        redis_prefix: str = "rt",
        max_concurrency: int = 20,
        retry_delay_seconds: int = 5,
    ):
        """Listen to a Mongo change stream and keep Redis up to date.

        Design overview
        ---------------
        - Change events arrive as a stream.  We extract the conversationId from
          each event and put it on a queue.
        - A ``pending`` set deduplicates: if a conversation is already queued or
          being processed, we mark it ``dirty`` instead of re-queuing.
        - A batch consumer drains the queue with a short debounce window so
          nearby changes are grouped together.
        - The accumulated batch is sliced into sub-batches of CDC_BATCH_PROCESS_SIZE
          (default 50).  Each sub-batch does ONE Mongo $in query and ONE Redis
          pipeline write.  This is the key fix for burst-load timeouts: instead
          of one huge operation for 400 conversations, we do eight small ones of 50.
        - A ``batch_sem`` (CDC_MAX_BATCHES_IN_FLIGHT, default 2) limits how many
          sub-batch tasks run concurrently so we never flood Redis with simultaneous
          pipelines.
        - After each sub-batch completes, any conversation marked dirty is re-queued
          so its latest state is not missed.
        """

        logger = getattr(app.state, "logger", None)

        class _RequestProxy:
            def __init__(self, app_ref):
                self.app = app_ref

        request_proxy = _RequestProxy(app)

        async def _log(level_fn, msg: str):
            try:
                if level_fn:
                    await level_fn(msg)
            except Exception:
                pass

        # How many conversationIds to include in one Mongo $in + Redis pipeline round.
        # Keeping this small is the primary defence against Redis socket timeouts.
        batch_process_size = int(os.getenv("CDC_BATCH_PROCESS_SIZE", "50"))
        # How many sub-batch tasks may run at the same time.
        max_batches_in_flight = int(os.getenv("CDC_MAX_BATCHES_IN_FLIGHT", "2"))
        # How long to wait after the first queued ID before firing a batch.
        batch_window_ms = int(os.getenv("CDC_BATCH_WINDOW_MS", "1500"))
        batch_window = max(0.05, batch_window_ms / 1000)

        while True:
            # ---- Outer restart loop ----------------------------------------
            try:
                target_db = dbname or (cls._config.get("database_name") if cls._config else None)
                if not target_db:
                    config = ConfigReader().get_mongo_config()
                    target_db = config.get("database_name")

                collection = await cls.get_database_collection(target_db, collectionname)
                pipeline = [
                    {"$match": {"operationType": {"$in": ["insert", "update", "replace"]}}}
                ]

                await _log(
                    logger.info if logger else None,
                    f"CDC stream: watching {target_db}.{collectionname} for changes",
                )

                from app.core.redis_connect import AWSRedis
                from app.service.realtime_metrics import RealtimeMetrics

                redis_client = AWSRedis()
                metrics = RealtimeMetrics(
                    redis_client=redis_client,
                    logger=logger,
                    redis_key_prefix=redis_prefix,
                )

                queue: asyncio.Queue[str] = asyncio.Queue()
                pending: set[str] = set()
                dirty: set[str] = set()
                state_lock = asyncio.Lock()
                sem = asyncio.Semaphore(max(1, max_concurrency))
                batch_sem = asyncio.Semaphore(max(1, max_batches_in_flight))
                active_batch_tasks: set[asyncio.Task] = set()

                async def process_sub_batch(sub_batch: list[str]):
                    """Process one small slice of conversations end-to-end.

                    One Mongo $in query → aggregate each conversation → one Redis
                    pipeline write.  Keeping sub-batches small (default 50) is
                    what prevents Redis pipeline timeouts under burst load.
                    """
                    try:
                        docs_by_cid = await cls.get_documents_grouped_by_conversation_ids(
                            request_proxy, target_db, collectionname, sub_batch
                        )
                        existing_by_cid = await redis_client.get_interactions_batch(
                            sub_batch, key_prefix=redis_prefix
                        )
                    except Exception as fetch_exc:
                        await _log(
                            logger.error if logger else None,
                            f"CDC sub-batch fetch error ({len(sub_batch)} convs): {fetch_exc}",
                        )
                        # Release pending for these IDs so they can be retried later.
                        async with state_lock:
                            for cid in sub_batch:
                                pending.discard(cid)
                        return

                    writes: dict[str, dict] = {}

                    async def handle(cid: str):
                        async with sem:
                            try:
                                docs = docs_by_cid.get(cid) or []
                                if not docs:
                                    return
                                aggregated = metrics._build_aggregated_payload(docs)
                                if not aggregated:
                                    return
                                merged = metrics._merge_with_existing(
                                    existing_by_cid.get(cid), aggregated
                                )
                                writes[cid] = merged
                            except Exception as w_exc:
                                await _log(
                                    logger.error if logger else None,
                                    f"CDC worker error for {cid}: {w_exc}",
                                )
                            finally:
                                requeue = False
                                async with state_lock:
                                    if cid in dirty:
                                        dirty.discard(cid)
                                        requeue = True
                                    else:
                                        pending.discard(cid)
                                if requeue:
                                    await queue.put(cid)

                    tasks = [asyncio.create_task(handle(cid)) for cid in sub_batch]
                    if tasks:
                        await asyncio.gather(*tasks)

                    if writes:
                        try:
                            await redis_client.write_interactions_batch(
                                writes, key_prefix=redis_prefix
                            )
                        except Exception as write_exc:
                            await _log(
                                logger.error if logger else None,
                                f"CDC Redis write error ({len(writes)} convs): {write_exc}",
                            )
                            # Don't crash — the dirty/requeue mechanism means
                            # the next CDC event for these conversations will retry.

                async def process_batch(batch: set[str]):
                    """Slice the accumulated batch into sub-batches and run them."""
                    batch_list = list(batch)
                    if logger:
                        try:
                            await logger.debug(
                                f"CDC processing {len(batch_list)} conversation(s) "
                                f"in sub-batches of {batch_process_size}"
                            )
                        except Exception:
                            pass

                    for start in range(0, len(batch_list), batch_process_size):
                        sub = batch_list[start : start + batch_process_size]
                        await process_sub_batch(sub)
                        # Yield between sub-batches so the stream reader and other
                        # coroutines stay responsive.
                        await asyncio.sleep(0)

                async def batch_consumer():
                    while True:
                        cid = await queue.get()
                        batch = {cid}
                        queue.task_done()
                        start = time_module.monotonic()
                        # Collect more IDs that arrive within the debounce window.
                        while True:
                            remaining = batch_window - (time_module.monotonic() - start)
                            if remaining <= 0:
                                break
                            try:
                                nxt = await asyncio.wait_for(queue.get(), timeout=remaining)
                                batch.add(nxt)
                                queue.task_done()
                            except asyncio.TimeoutError:
                                break

                        async def _run_batch(b: set[str]):
                            async with batch_sem:
                                await process_batch(b)

                        batch_task = asyncio.create_task(_run_batch(batch))
                        active_batch_tasks.add(batch_task)
                        batch_task.add_done_callback(active_batch_tasks.discard)

                consumer_task = asyncio.create_task(batch_consumer())
                await _log(
                    logger.info if logger else None,
                    f"CDC consumer started (window={batch_window_ms}ms, "
                    f"sub_batch={batch_process_size}, "
                    f"max_in_flight={max_batches_in_flight}) "
                    f"for {target_db}.{collectionname}",
                )

                async with collection.watch(pipeline, full_document="updateLookup") as stream:
                    async for change in stream:
                        full_doc = (
                            change.get("fullDocument") if isinstance(change, dict) else None
                        )
                        if not full_doc:
                            continue
                        cid = (
                            full_doc.get("conversationId")
                            if isinstance(full_doc, dict)
                            else None
                        )
                        if not cid:
                            continue
                        async with state_lock:
                            if cid in pending:
                                dirty.add(cid)
                                continue
                            pending.add(cid)
                        await queue.put(cid)

            except (PyMongoError, Exception) as exc:
                await _log(
                    logger.error if logger else None,
                    f"CDC stream error for {collectionname}: {exc}",
                )
                await asyncio.sleep(max(1, retry_delay_seconds))
            finally:
                try:
                    c = locals().get("consumer_task")
                    if c:
                        c.cancel()
                except Exception:
                    pass
                try:
                    for t in list(locals().get("active_batch_tasks", set())):
                        t.cancel()
                except Exception:
                    pass

    @classmethod
    async def dump_redis_prefix_to_mongo(
        cls,
        app,
        *,
        dbname: str | None = None,
        redis_prefix: str = "rt",
        target_collection: str = "conversation_anirudh_download2",
        ttl_extension_seconds: int = 900,
        delete_after_dump: bool = True,
        batch_size: int = 1000,
    ):
        """Snapshot all Redis keys under prefix into a Mongo collection."""
        logger = getattr(app.state, "logger", None)

        def _now():
            return datetime.now(timezone.utc)

        try:
            from app.core.redis_connect import AWSRedis

            redis_client = AWSRedis()
            data = await redis_client.get_all_with_prefix(
                redis_prefix,
                extend_ttl_seconds=ttl_extension_seconds,
                delete_after=False,
            )

            if not data:
                if logger:
                    await logger.info(f"Redis dump: no keys found for prefix {redis_prefix}")
                return 0

            target_db = dbname or (cls._config.get("database_name") if cls._config else None)
            if not target_db:
                config = ConfigReader().get_mongo_config()
                target_db = config.get("database_name")

            collection = await cls.get_database_collection(target_db, target_collection)

            docs = []
            dumped_at = _now()
            for full_key, payload in data.items():
                doc = payload if isinstance(payload, dict) else {"value": payload}
                doc = dict(doc)
                doc["_redis_key"] = full_key
                doc["_dumped_at_utc"] = dumped_at
                docs.append(doc)

            inserted = 0
            for i in range(0, len(docs), batch_size):
                batch = docs[i : i + batch_size]
                try:
                    res = await collection.insert_many(batch, ordered=False)
                    inserted += len(res.inserted_ids)
                except Exception as batch_exc:
                    if logger:
                        await logger.error(f"Redis dump batch insert failed: {batch_exc}")

            if delete_after_dump:
                try:
                    await redis_client.delete_keys(list(data.keys()))
                except Exception as del_exc:
                    if logger:
                        await logger.error(f"Redis dump cleanup failed: {del_exc}")

            if logger:
                await logger.info(
                    f"Redis dump complete: {inserted} docs from prefix {redis_prefix} "
                    f"into {target_db}.{target_collection}"
                )
            return inserted

        except Exception as exc:
            if logger:
                await logger.error(f"Redis dump failed for prefix {redis_prefix}: {exc}")
            return 0

    @classmethod
    async def schedule_daily_redis_dump(
        cls,
        app,
        *,
        redis_prefix: str = "rt",
        target_collection: str = "conversation_anirudh_download2",
        lead_seconds: int = 30,
        ttl_extension_seconds: int = 900,
    ):
        """Sleep until just before midnight UTC, then dump Redis to Mongo. Repeats daily."""
        logger = getattr(app.state, "logger", None)

        def seconds_until_next_midnight_utc():
            now = datetime.now(timezone.utc)
            next_midnight = (now + timedelta(days=1)).replace(
                hour=0, minute=0, second=0, microsecond=0
            )
            return max(1, int((next_midnight - now).total_seconds()))

        while True:
            try:
                sleep_for = max(1, seconds_until_next_midnight_utc() - max(0, lead_seconds))
                if logger:
                    await logger.info(
                        f"Redis dump scheduler sleeping {sleep_for}s "
                        f"(prefix={redis_prefix}, collection={target_collection})"
                    )
                await asyncio.sleep(sleep_for)
                if logger:
                    await logger.info(
                        f"Starting scheduled Redis dump "
                        f"(prefix={redis_prefix}, collection={target_collection})"
                    )
                await cls.dump_redis_prefix_to_mongo(
                    app,
                    redis_prefix=redis_prefix,
                    target_collection=target_collection,
                    ttl_extension_seconds=ttl_extension_seconds,
                    delete_after_dump=True,
                )
            except Exception as exc:
                if logger:
                    await logger.error(f"Scheduled Redis dump failed: {exc}")
            finally:
                await asyncio.sleep(5)

# ---------------------------------------------------------------------------
# This module is the business brain of the service.
#
# It does five things:
#   1. Parse Mongo documents into a typed, comparable shape.
#   2. Sort those documents into business order.
#   3. Compute one final aggregated view per conversation.
#   4. Merge that view with any prior Redis value (sticky flags, running totals).
#   5. Persist the result to Redis.
#
# There are two call paths into this module:
#
#   ONE-TIME LOAD (startup bulk import)
#     Called by process_one_time_load() at the bottom of this file.
#     Uses _process_initial_load_batches_of_conversations() which skips the
#     per-conversation Redis read because the load is treated as a clean slate.
#
#   CDC (live change stream)
#     Called by connectionpool.start_cdc_stream().
#     Uses _build_aggregated_payload() + _merge_with_existing() directly so the
#     CDC loop can batch reads and writes without going through the one-at-a-time
#     _process_conversation_group() method.
#
# _process_conversation_group() still exists and is correct; it is used by
# legacy callers and test code.  CDC bypasses it intentionally for performance.
# ---------------------------------------------------------------------------

def _parse_iso(ts: Any) -> Optional[datetime]:
    """Convert string/datetime to a timezone-aware UTC datetime, or None."""
    if ts is None:
        return None
    if isinstance(ts, datetime):
        return ts if ts.tzinfo else ts.replace(tzinfo=timezone.utc)
    if isinstance(ts, str):
        s = ts.strip()
        if not s:
            return None
        if s.endswith("Z"):
            s = s[:-1] + "+00:00"
        try:
            parsed = datetime.fromisoformat(s)
            return parsed if parsed.tzinfo else parsed.replace(tzinfo=timezone.utc)
        except Exception:
            return None
    return None


def _status_events(seq: Any) -> List[tuple]:
    """Turn a status list into (status_name, timestamp) pairs.

    Example input:  [{"INITIATED": "2026-03-23T05:23:43Z"}, {"TRANSFERRED": "..."}]
    Example output: [("INITIATED", datetime(...)), ("TRANSFERRED", datetime(...))]
    """
    if not seq or not isinstance(seq, list):
        return []
    events: List[tuple] = []
    for item in seq:
        if not isinstance(item, dict):
            continue
        for key, value in item.items():
            ts = _parse_iso(value)
            if ts:
                events.append((str(key).strip().upper(), ts))
            break
    return events


def _timed_channel_events(seq: Any, default_ts: Optional[datetime] = None) -> List[tuple]:
    """Turn the `channel` field into (timestamp, channel_name) pairs."""
    if not seq or not isinstance(seq, list):
        return []
    events: List[tuple] = []
    for item in seq:
        if isinstance(item, dict):
            for key, value in item.items():
                ts = _parse_iso(value) or default_ts
                if ts:
                    events.append((ts, str(key).strip().upper()))
                break
        elif isinstance(item, str) and default_ts:
            channel_name = item.strip().upper()
            if channel_name:
                events.append((default_ts, channel_name))
    return events


def _doc_start_timestamp(doc: Dict[str, Any]) -> datetime:
    """Earliest meaningful timestamp in a document (used for chronological sort)."""
    candidates: List[datetime] = []
    for raw_ts in (doc.get("createdDateTime"), doc.get("timeStamp")):
        parsed = _parse_iso(raw_ts)
        if parsed:
            candidates.append(parsed)
    for field in ("conversationStatus", "interactionStatus", "channel", "details"):
        if field == "channel":
            candidates.extend(ts for ts, _ in _timed_channel_events(doc.get(field)))
        else:
            candidates.extend(ts for _, ts in _status_events(doc.get(field)))
    if candidates:
        return min(candidates)
    fallback = _parse_iso(doc.get("lastModifiedDateTime"))
    return fallback or datetime.min.replace(tzinfo=timezone.utc)


def _doc_end_timestamp(doc: Dict[str, Any]) -> datetime:
    """Latest meaningful timestamp in a document (used for 'freshest snapshot' logic)."""
    candidates: List[datetime] = []
    for raw_ts in (doc.get("createdDateTime"), doc.get("timeStamp"), doc.get("lastModifiedDateTime")):
        parsed = _parse_iso(raw_ts)
        if parsed:
            candidates.append(parsed)
    for field in ("conversationStatus", "interactionStatus", "channel", "details"):
        if field == "channel":
            candidates.extend(ts for ts, _ in _timed_channel_events(doc.get(field)))
        else:
            candidates.extend(ts for _, ts in _status_events(doc.get(field)))
    return max(candidates) if candidates else datetime.min.replace(tzinfo=timezone.utc)

def _last_key_from_status_list(seq: Any) -> Optional[str]:
    if not seq or not isinstance(seq, list):
        return None
    last = seq[-1]
    if isinstance(last, dict):
        for k in last.keys():
            return str(k).upper()
    if isinstance(last, str):
        return last.strip().upper() or None
    return None


def _has_status(seq: Any, targets: set) -> bool:
    if not seq or not isinstance(seq, list) or not targets:
        return False
    targets_up = {t.upper() for t in targets}
    for item in seq:
        if isinstance(item, dict):
            for k in item.keys():
                if str(k).strip().upper() in targets_up:
                    return True
        elif isinstance(item, str):
            if item.strip().upper() in targets_up:
                return True
    return False

    try:

    if not layer_name:
        return None
    layer_up = str(layer_name).upper()
    convo_type_up = str(conversation_type).strip().upper() if conversation_type else None
    if layer_up == "BOT":

    if layer_up == "GENESYS":
        return "CHATAGENTCLOUD" if convo_type_up == "CHAT" else "VOICEAGENTCLOUD"
    if layer_up in {"IVR", "VOL", "CBOL"}:
        return layer_up
    return _channel_name_for_layer(layer_up)

class RealtimeMetrics:
    def __init__(
        self,
        redis_client: AWSRedis | None = None,
        logger: Any | None = None,
        redis_key_prefix: str | None = None,
    ):
        self.redis_client = redis_client or AWSRedis()
        self.logger = logger
        self.redis_key_prefix = redis_key_prefix

    # ------------------------------------------------------------------
    # Redis write helpers
    # ------------------------------------------------------------------

    async def _write_aggregated(self, conversation_id: str, aggregated: Dict[str, Any]):
        """Write one aggregated conversation to Redis (single-key path)."""
        try:
            if self.logger:
                try:
                    await self.logger.debug(
                        f"_write_aggregated: id={conversation_id}, "
                        f"keys={list(aggregated.keys())}"
                    )
                except Exception:
                    pass

            await self.redis_client.write_interaction(
                conversation_id,
                aggregated,
                key_prefix=self.redis_key_prefix,
            )
        except Exception as exc:
            msg = f"Failed to write aggregated conversation {conversation_id}: {exc}"
            if self.logger:
                try:
                    await self.logger.error(msg)
                except Exception:
                    print(msg)
            else:
                print(msg)

    # ------------------------------------------------------------------
    # Merge logic
    # ------------------------------------------------------------------

    def _merge_with_existing(
        self, existing: Dict[str, Any] | None, new: Dict[str, Any]
    ) -> Dict[str, Any]:
        """Merge freshly computed metrics with any prior Redis value.

        Rules:
        - Latest fields  — always overwrite with the newest snapshot value.
        - Sticky flags   — once they become 1, they never go back to 0.
        - Terminal flags — reflect the latest recomputation.
        - Time fields    — take max() to avoid double-counting on reprocess.
        - Intent/history — union, order-preserving.
        """
        if not existing:
            return new

        merged = dict(existing)

        latest_fields = [
            "conversation_state",
            "active_layer",
            "bot_in_progress",
            "ivr_in_progress",
            "genesys_in_progress",
            "conversation_type",
            "conversation_in_queue",
            "last_seen_at",
        ]
        for f in latest_fields:
            merged[f] = new.get(f, merged.get(f))

        def _earliest(existing_val, new_val):
            a = _parse_iso(existing_val)
            b = _parse_iso(new_val)
            if a and b:
                return a.isoformat() if a <= b else b.isoformat()
            return new_val or existing_val

        merged["first_seen_at"] = _earliest(
            existing.get("first_seen_at"), new.get("first_seen_at")
        )
        if merged.get("last_seen_at"):
            ts_last = _parse_iso(merged.get("last_seen_at"))
            if ts_last:
                merged["last_seen_at"] = ts_last.isoformat()

        sticky_flags = [
            "bot_closed",
            "ivr_closed",
            "genesys_closed",
            "genesys_conference_call",
            "voicebot_to_voiceagent",
            "voicebot_to_ivr",
            "chatbot_to_chatagent",
            "chatbot_to_ivr",
        ]
        for f in sticky_flags:
            merged[f] = int(bool(merged.get(f)) or bool(new.get(f)))

        latest_terminal_flags = [
            "bot_contained",
            "ivr_contained",
            "genesys_contained",
            "conversation_closed_ind",
            "conversation_ended_by_system",
            "conversation_ended_by_customer",
        ]
        for f in latest_terminal_flags:
            merged[f] = new.get(f, merged.get(f))

        time_fields = [
            "conversation_handling_time",
            "genesys_handling_time",
            "bot_handling_time",
            "ivr_handling_time",
        ]
        for f in time_fields:
            prev_val = merged.get(f) or 0
            new_val = new.get(f) or 0
            try:
                merged[f] = round(max(float(prev_val), float(new_val)), 2)
            except Exception:
                merged[f] = new_val or prev_val

        merged["interaction_handling_time"] = round(
            (merged.get("bot_handling_time") or 0)
            + (merged.get("ivr_handling_time") or 0)
            + (merged.get("genesys_handling_time") or 0),
            2,
        )

        merged["sentiment"] = (
            new.get("sentiment")
            if new.get("sentiment") is not None
            else merged.get("sentiment")
        )

        merged_intent: List[str] = []
        for item in merged.get("intent") or []:
            if item not in merged_intent:
                merged_intent.append(item)
        for item in new.get("intent") or []:
            if item not in merged_intent:
                merged_intent.append(item)
        merged["intent"] = merged_intent if merged_intent else None

        def _merge_ordered_list(existing_list, new_list):
            out: List[str] = []
            for src in new_list or []:
                if src not in out:
                    out.append(src)
            for src in existing_list or []:
                if src not in out:
                    out.append(src)
            return out if out else None

        merged["channel_path"] = new.get("channel_path") or existing.get("channel_path")
        merged["visited_layers"] = _merge_ordered_list(
            existing.get("visited_layers"), new.get("visited_layers")
        )
        merged["starting_layer"] = new.get("starting_layer") or existing.get("starting_layer")

        for f in ["conversationId", "market", "region"]:
            if f in new:
                merged[f] = new[f]

        return merged

    # ------------------------------------------------------------------
    # Core aggregation
    # ------------------------------------------------------------------

    def _build_aggregated_payload(self, docs: List[Dict[str, Any]]) -> Dict[str, Any] | None:
        """Build one Redis object from all Mongo docs of a conversation.

        This is the most important function in the file.  It answers:
        - what is the latest conversation state?
        - which layer handled it last?
        - did it close / get contained?
        - which channels were visited?
        - how long did each layer handle it?
        """
        if not docs:
            return None

        docs_sorted = sorted(docs, key=_doc_start_timestamp)

        first = docs_sorted[0]
        conversation_id_val = first.get("conversationId")
        market = first.get("market")
        region = first.get("region")

        latest_doc = max(docs_sorted, key=_doc_end_timestamp)
        conversation_type = latest_doc.get("conversationType")

        sentiment_doc = max(
            (d for d in docs_sorted if d.get("sentiment")),
            key=_timestamp_for_doc,
            default=None,
        )
        last_sentiment = (
            _last_object_key(sentiment_doc.get("sentiment")) if sentiment_doc else None
        )

        intent_history: List[str] = []
        for doc in docs_sorted:
            intent_history.extend(_collect_object_keys(doc.get("intent")))

        latest_status = _normalize_status(
            _last_key_from_status_list(latest_doc.get("conversationStatus"))
        )
        latest_layer = _layer_for_org(latest_doc.get("organization"))

        terminal_doc = None
        terminal_status = None
        for doc in docs_sorted:
            status = _normalize_status(
                _last_key_from_status_list(doc.get("conversationStatus"))
            )
            if status in {"CLOSED", "ENDED_BY_SYSTEM", "ENDED_BY_CUSTOMER", "ENDED"}:
                org_layer = _layer_for_org(doc.get("organization"))
                if org_layer in {"VOL", "CBOL"}:
                    if terminal_doc is None or _doc_end_timestamp(doc) > _doc_end_timestamp(
                        terminal_doc
                    ):
                        terminal_doc = doc
                        terminal_status = status

        if terminal_doc is not None:
            conversation_state = terminal_status
            active_layer = _layer_for_org(terminal_doc.get("organization"))
            state_doc = terminal_doc
        else:
            for doc in docs_sorted:
                status = _normalize_status(
                    _last_key_from_status_list(doc.get("conversationStatus"))
                )
                if _status_rank(status) > _status_rank(latest_status):
                    latest_status = status
                    latest_layer = _layer_for_org(doc.get("organization"))
            conversation_state = latest_status
            active_layer = latest_layer
            state_doc = latest_doc

        layer_latest_status: dict = {}
        for doc in docs_sorted:
            layer = _layer_for_org(doc.get("organization"))
            if not layer:
                continue
            ts = _doc_end_timestamp(doc)
            status_raw = _last_key_from_status_list(
                doc.get("conversationStatus")
            ) or _last_key_from_status_list(doc.get("interactionStatus"))
            status = _normalize_status(status_raw)
            if layer not in layer_latest_status or ts >= layer_latest_status[layer][0]:
                layer_latest_status[layer] = (ts, status)

        def _in_progress(layer: str) -> int:
            status = layer_latest_status.get(layer, (None, None))[1]
            return 1 if status in {"IN_PROGRESS", "INITIATED"} else 0

        bot_in_progress = _in_progress("BOT")
        ivr_in_progress = _in_progress("IVR")
        genesys_in_progress = _in_progress("GENESYS")

        closed_like_statuses = {"CLOSED", "ENDED_BY_SYSTEM", "ENDED_BY_CUSTOMER", "ENDED"}
        if conversation_state in closed_like_statuses:
            bot_in_progress = 0
            ivr_in_progress = 0
            genesys_in_progress = 0

        bot_closed = 0
        ivr_closed = 0
        genesys_closed = 0
        bot_contained = 0
        ivr_contained = 0
        genesys_contained = 0
        genesys_conference_call = 0
        visited_layers: List[str] = []
        channel_events: List[tuple] = []
        manager_channel_events: List[tuple] = []
        interactive_layer_events: List[tuple] = []
        interactive_layers_present: set = set()
        synthetic_interactive_channel_events: List[tuple] = []

        for doc in docs_sorted:
            doc_layer = _layer_for_org(doc.get("organization"))
            if doc_layer and doc_layer not in visited_layers:
                visited_layers.append(doc_layer)

            timed_channels = _timed_channel_events(
                doc.get("channel"), _doc_start_timestamp(doc)
            )
            channel_events.extend(timed_channels)

            if doc_layer in {"BOT", "IVR", "GENESYS"}:
                doc_start_ts = _doc_start_timestamp(doc)
                interactive_layer_events.append((doc_start_ts, doc_layer))
                interactive_layers_present.add(doc_layer)
                has_matching_channel = any(
                    _layer_for_channel(cn) == doc_layer for _, cn in timed_channels
                )
                if not has_matching_channel:
                    canonical_channel = _canonical_channel_for_layer(doc_layer, conversation_type)
                    if canonical_channel:
                        synthetic_interactive_channel_events.append(
                            (doc_start_ts, canonical_channel)
                        )

            if doc_layer in {"VOL", "CBOL"}:
                for ts, channel_name in timed_channels:
                    manager_channel_events.append(
                        (ts, channel_name, _layer_for_channel(channel_name) or channel_name)
                    )

            conv_status_list = doc.get("conversationStatus")
            has_closed = _has_status(conv_status_list, {"CLOSED"})
            has_closed_or_transferred = _has_status(
                conv_status_list, {"CLOSED", "TRANSFERRED"}
            )
            has_conference = _has_status(conv_status_list, {"CONFERENCE"}) or _has_status(
                doc.get("interactionStatus"), {"CONFERENCE", "CONSULT_CONFERENCE"}
            )

            if doc_layer == "BOT":
                if has_closed_or_transferred:
                    bot_closed = 1
                if has_closed:
                    bot_contained = 1
            if doc_layer == "IVR":
                if has_closed_or_transferred:
                    ivr_closed = 1
                if has_closed:
                    ivr_contained = 1
            if doc_layer == "GENESYS":
                if has_closed_or_transferred:
                    genesys_closed = 1
                if has_conference:
                    genesys_conference_call = 1

        channel_events.sort(key=lambda x: x[0])
        manager_channel_events.sort(key=lambda x: x[0])
        interactive_layer_events.sort(key=lambda x: x[0])
        synthetic_interactive_channel_events.sort(key=lambda x: x[0])

        actual_channel_events: List[tuple] = []
        for ts, channel_name in channel_events:
            mapped_layer = _layer_for_channel(channel_name)
            if (
                mapped_layer in {"BOT", "IVR", "GENESYS"}
                and mapped_layer not in interactive_layers_present
            ):
                continue
            actual_channel_events.append((ts, channel_name))

        actual_channel_events.extend(synthetic_interactive_channel_events)
        actual_channel_events.sort(key=lambda x: x[0])

        channel_path: List[str] = []
        for _, channel_name in actual_channel_events:
            if not channel_path or channel_path[-1] != channel_name:
                channel_path.append(channel_name)

        terminal_channel_name = _channel_name_for_layer(active_layer)
        if terminal_channel_name and terminal_channel_name in {"VOL", "CBOL"}:
            if not channel_path or channel_path[-1] != terminal_channel_name:
                channel_path.append(terminal_channel_name)

        starting_layer = None
        first_layer = _layer_for_org(first.get("organization"))
        if first_layer == "VOL":
            starting_layer = "VOICE"
        elif first_layer == "CBOL":
            starting_layer = "CHAT"
        elif conversation_type:
            convo_type = str(conversation_type).strip().upper()
            if convo_type in {"VOICE", "CHAT"}:
                starting_layer = convo_type

        channel_seq = channel_path[:]
        voicebot_to_voiceagent = 0
        voicebot_to_ivr = 0
        chatbot_to_chatagent = 0
        chatbot_to_ivr = 0
        if channel_seq:
            def _ordered_any(before_set: set, after_set: set) -> bool:
                before_idx = min(
                    (channel_seq.index(b) for b in before_set if b in channel_seq),
                    default=None,
                )
                if before_idx is None:
                    return False
                for candidate in after_set:
                    if candidate in channel_seq and channel_seq.index(candidate) > before_idx:
                        return True
                return False

            voice_bot_channels = {"VOICEBOT", "BOT", "SIERRA"}
            chat_bot_channels = {"CHATBOT", "BOT", "SIERRA"}
            voice_agent_channels = {"VOICEAGENTCLOUD", "VOICEAGENTONPREM", "GENESYS"}
            if starting_layer == "VOICE":
                if _ordered_any(voice_bot_channels, voice_agent_channels):
                    voicebot_to_voiceagent = 1
                if _ordered_any(voice_bot_channels, {"IVR"}):
                    voicebot_to_ivr = 1
            if starting_layer == "CHAT":
                if _ordered_any(chat_bot_channels, {"CHATAGENTCLOUD", "CHATAGENTONPREM"}):
                    chatbot_to_chatagent = 1
                if _ordered_any(chat_bot_channels, {"IVR"}):
                    chatbot_to_ivr = 1

        if voicebot_to_voiceagent or voicebot_to_ivr or chatbot_to_chatagent or chatbot_to_ivr:
            bot_closed = 1

        end_status = terminal_status or latest_status
        end_layer = active_layer or latest_layer
        conversation_ended_by_system = 1 if end_status == "ENDED_BY_SYSTEM" else 0
        conversation_ended_by_customer = 1 if end_status == "ENDED_BY_CUSTOMER" else 0

        actual_interactive_layers: List[str] = []
        for _, channel_name in actual_channel_events:
            mapped_layer = _layer_for_channel(channel_name)
            if mapped_layer in {"BOT", "IVR", "GENESYS"}:
                actual_interactive_layers.append(mapped_layer)

        contained_layer = None
        if end_layer in {"VOL", "CBOL"} and end_status in closed_like_statuses:
            for layer_name in reversed(actual_interactive_layers):
                if layer_name in {"BOT", "IVR", "GENESYS"}:
                    contained_layer = layer_name
                    break
        elif end_layer in {"BOT", "IVR", "GENESYS"} and end_status in closed_like_statuses:
            contained_layer = end_layer

        bot_contained = 1 if contained_layer == "BOT" else 0
        ivr_contained = 1 if contained_layer == "IVR" else 0
        genesys_contained = 1 if contained_layer == "GENESYS" else 0

        conversation_closed_ind = (
            1 if end_layer in {"VOL", "CBOL"} and end_status in closed_like_statuses else 0
        )
        if bot_contained or ivr_contained or genesys_contained:
            conversation_closed_ind = 1

        earliest_seen = min(
            (_doc_start_timestamp(doc) for doc in docs_sorted),
            default=datetime.min.replace(tzinfo=timezone.utc),
        )
        latest_seen = _doc_end_timestamp(state_doc)
        conversation_handling_time = max(
            0.0, (latest_seen - earliest_seen).total_seconds()
        )

        bot_handling_time = 0.0
        ivr_handling_time = 0.0
        genesys_handling_time = 0.0
        bot_orgs = {
            "VOICEBOT", "CHATBOT", "BOT", "SIERRA",
            "CIVR_SIERRA", "CBOL_BOT", "VOL_BOT", "CHAT_BOT", "VOICE_BOT",
        }
        ivr_orgs = {"IVR"}
        genesys_orgs = {
            "GENESYS", "CBOL_GENESYS", "VOL_GENESYS",
            "CHATAGENTCLOUD", "CHATAGENTONPREM",
            "VOICEAGENTCLOUD", "VOICEAGENTONPREM",
        }

        def _next_manager_handoff(layer_name: str, start_ts: datetime) -> Optional[datetime]:
            for ts, _, mapped_layer in manager_channel_events:
                if ts <= start_ts:
                    continue
                if mapped_layer != layer_name:
                    return ts
            return None

        interactive_docs: List[tuple] = []
        for doc in docs_sorted:
            layer_name = _layer_for_org(doc.get("organization"))
            if layer_name in {"BOT", "IVR", "GENESYS"}:
                interactive_docs.append((_doc_start_timestamp(doc), layer_name, doc))

        for index, (doc_start_ts, doc_layer, doc) in enumerate(interactive_docs):
            org = doc.get("organization")
            org_up = str(org).upper() if org else ""
            own_terminal_candidates: List[datetime] = []
            for status, ts in (
                _status_events(doc.get("conversationStatus"))
                + _status_events(doc.get("interactionStatus"))
            ):
                if _normalize_status(status) in {
                    "TRANSFERRED", "CLOSED", "ENDED_BY_SYSTEM", "ENDED_BY_CUSTOMER", "ENDED"
                }:
                    own_terminal_candidates.append(ts)

            next_layer_start = None
            for next_start_ts, next_layer, _ in interactive_docs[index + 1 :]:
                if next_layer != doc_layer:
                    next_layer_start = next_start_ts
                    break

            candidate_end_times = [ts for ts in own_terminal_candidates if ts > doc_start_ts]
            manager_handoff_ts = _next_manager_handoff(doc_layer, doc_start_ts)
            if manager_handoff_ts and manager_handoff_ts > doc_start_ts:
                candidate_end_times.append(manager_handoff_ts)
            if next_layer_start and next_layer_start > doc_start_ts:
                candidate_end_times.append(next_layer_start)

            if not candidate_end_times:
                fallback_end = _doc_end_timestamp(doc)
                if fallback_end > doc_start_ts:
                    candidate_end_times.append(fallback_end)

            if not candidate_end_times:
                continue

            doc_end_ts = min(candidate_end_times)
            secs = max(0.0, (doc_end_ts - doc_start_ts).total_seconds())

            inferred_closed = bool(own_terminal_candidates) or (
                manager_handoff_ts is not None
            ) or (next_layer_start is not None)
            if inferred_closed:
                if org_up in bot_orgs:
                    bot_closed = 1
                if org_up in ivr_orgs:
                    ivr_closed = 1
                if org_up in genesys_orgs:
                    genesys_closed = 1

            if org_up in bot_orgs:
                bot_handling_time += secs
            if org_up in ivr_orgs:
                ivr_handling_time += secs
            if org_up in genesys_orgs:
                genesys_handling_time += secs

        transfer_from_vol_cbol_ts = None
        in_progress_after_transfer = False
        for doc in docs_sorted:
            layer = _layer_for_org(doc.get("organization"))
            if layer in {"VOL", "CBOL"}:
                for status, ts in _status_events(doc.get("conversationStatus")):
                    if _normalize_status(status) == "TRANSFERRED":
                        if transfer_from_vol_cbol_ts is None or ts < transfer_from_vol_cbol_ts:
                            transfer_from_vol_cbol_ts = ts
            if not transfer_from_vol_cbol_ts:
                continue
            all_events = _status_events(doc.get("conversationStatus")) + _status_events(
                doc.get("interactionStatus")
            )
            if any(
                _normalize_status(s) == "IN_PROGRESS" and ts > transfer_from_vol_cbol_ts
                for s, ts in all_events
            ):
                in_progress_after_transfer = True
                break

        conversation_in_queue = (
            1 if transfer_from_vol_cbol_ts and not in_progress_after_transfer else 0
        )
        interaction_handling_time = round(
            bot_handling_time + ivr_handling_time + genesys_handling_time, 2
        )

        return {
            "conversationId": conversation_id_val,
            "market": market,
            "region": region,
            "first_seen_at": earliest_seen.isoformat(),
            "last_seen_at": latest_seen.isoformat(),
            "conversation_type": conversation_type,
            "conversation_state": conversation_state,
            "active_layer": active_layer,
            "starting_layer": starting_layer,
            "sentiment": last_sentiment,
            "intent": intent_history if intent_history else None,
            "visited_layers": visited_layers if visited_layers else None,
            "channel_path": channel_path if channel_path else None,
            "conversation_closed_ind": int(conversation_closed_ind),
            "conversation_in_queue": int(conversation_in_queue),
            "bot_in_progress": int(bot_in_progress),
            "ivr_in_progress": int(ivr_in_progress),
            "genesys_in_progress": int(genesys_in_progress),
            "bot_closed": int(bot_closed),
            "ivr_closed": int(ivr_closed),
            "genesys_closed": int(genesys_closed),
            "voicebot_to_voiceagent": int(voicebot_to_voiceagent),
            "voicebot_to_ivr": int(voicebot_to_ivr),
            "chatbot_to_chatagent": int(chatbot_to_chatagent),
            "chatbot_to_ivr": int(chatbot_to_ivr),
            "genesys_conference_call": int(genesys_conference_call),
            "conversation_ended_by_system": int(conversation_ended_by_system),
            "conversation_ended_by_customer": int(conversation_ended_by_customer),
            "ivr_contained": int(ivr_contained),
            "bot_contained": int(bot_contained),
            "genesys_contained": int(genesys_contained),
            "conversation_handling_time": round(conversation_handling_time, 2),
            "interaction_handling_time": interaction_handling_time,
            "genesys_handling_time": round(genesys_handling_time, 2),
            "bot_handling_time": round(bot_handling_time, 2),
            "ivr_handling_time": round(ivr_handling_time, 2),
        }

    # Process paths
    # ------------------------------------------------------------------

    async def _process_conversation_group(
        self, conversation_id: str, docs: List[Dict[str, Any]]
    ) -> None:
        """Aggregate one conversation and persist to Redis (single-conversation path).

        Used by the one-time load legacy path and standalone callers.
        CDC does NOT call this method — it calls _build_aggregated_payload() +
        _merge_with_existing() directly and batch-writes the results.
        """
        aggregated = self._build_aggregated_payload(docs)
        if not aggregated:
            return
        conversation_id_val = aggregated.get("conversationId") or conversation_id

        if self.logger:
            try:
                await self.logger.debug(
                    f"Persisting conversation: id={conversation_id_val}, "
                    f"docs={len(docs)}"
                )
            except Exception:
                pass

        existing = None
        try:
            existing = await self.redis_client.get_interaction(
                conversation_id_val, key_prefix=self.redis_key_prefix
            )
        except Exception:
            existing = None

        merged = self._merge_with_existing(existing, aggregated)
        await self._write_aggregated(conversation_id_val, merged)

    async def _build_conversation_payload(
        self,
        conversation_id: str,
        docs: List[Dict[str, Any]],
        *,
        merge_with_existing: bool,
    ) -> tuple | None:
        aggregated = self._build_aggregated_payload(docs)
        if not aggregated:
            return None
        conversation_id_val = aggregated.get("conversationId") or conversation_id
        payload = aggregated
        if merge_with_existing:
            existing = None
            try:
                existing = await self.redis_client.get_interaction(
                    conversation_id_val, key_prefix=self.redis_key_prefix
                )
            except Exception:
                existing = None
            payload = self._merge_with_existing(existing, aggregated)
        return conversation_id_val, payload

    async def _process_batches_of_conversations(
        self, convo_items: Iterable[tuple], *, concurrency: int
    ) -> int:
        """Legacy batch path used by the one-time load (merge_with_existing=True).

        Each conversation does an individual Redis read before writing.
        For the initial load fast path use _process_initial_load_batches_of_conversations().
        """
        sem = asyncio.Semaphore(max(1, concurrency))
        tasks: List[asyncio.Task] = []
        convo_list = list(convo_items)
        if self.logger:
            try:
                await self.logger.info(
                    f"Processing batch of {len(convo_list)} conversation(s)"
                )
            except Exception:
                pass

        async def _wrap(convo_id, docs):
            async with sem:
                await self._process_conversation_group(convo_id, docs)

        for convo_id, docs in convo_list:
            tasks.append(asyncio.create_task(_wrap(convo_id, docs)))
        if tasks:
            await asyncio.gather(*tasks)
        return len(tasks)

    async def _process_initial_load_batches_of_conversations(
        self,
        convo_items: Iterable[tuple],
        *,
        concurrency: int,
        redis_write_batch_size: int,
    ) -> int:
        """One-time load fast path.

        Differences from the legacy CDC path:
        - No per-conversation Redis read before writing (clean-slate assumption).
        - All writes are batched into pipelines of redis_write_batch_size.
        - On pipeline failure the batch is split recursively until it succeeds or
          reaches a single-item fallback.
        """
        sem = asyncio.Semaphore(max(1, concurrency))
        tasks: List[asyncio.Task] = []
        writes: Dict[str, Dict[str, Any]] = {}
        writes_lock = asyncio.Lock()

        convo_list = list(convo_items)
        if self.logger:
            try:
                await self.logger.info(
                    f"Initial-load batch: {len(convo_list)} conversation(s)"
                )
            except Exception:
                pass

        async def _wrap(convo_id, docs):
            async with sem:
                payload = await self._build_conversation_payload(
                    convo_id, docs, merge_with_existing=False
                )
                if not payload:
                    return
                conversation_id_val, aggregated = payload
                async with writes_lock:
                    writes[conversation_id_val] = aggregated

        for convo_id, docs in convo_list:
            tasks.append(asyncio.create_task(_wrap(convo_id, docs)))
        if tasks:
            await asyncio.gather(*tasks)

        if not writes:
            return 0

        items = list(writes.items())
        chunk_size = max(1, redis_write_batch_size)
        pause_seconds = max(
            0.0,
            float(os.getenv("ONE_TIME_REDIS_WRITE_PAUSE_MS", "25")) / 1000.0,
        )

        async def _write_chunk_with_fallback(chunk_items: List[tuple]) -> None:
            if not chunk_items:
                return
            chunk = dict(chunk_items)
            try:
                await self.redis_client.write_interactions_batch(
                    chunk, key_prefix=self.redis_key_prefix
                )
                return
            except Exception as exc:
                if len(chunk_items) == 1:
                    conversation_id_val, aggregated = chunk_items[0]
                    if self.logger:
                        try:
                            await self.logger.warning(
                                f"Fallback to single write for {conversation_id_val}: {exc}"
                            )
                        except Exception:
                            pass
                    await self.redis_client.write_interaction(
                        conversation_id_val,
                        aggregated,
                        key_prefix=self.redis_key_prefix,
                    )
                    return

                if self.logger:
                    try:
                        await self.logger.warning(
                            f"Batch write failed ({len(chunk_items)} convs); splitting: {exc}"
                        )
                    except Exception:
                        pass
                mid = len(chunk_items) // 2
                await _write_chunk_with_fallback(chunk_items[:mid])
                await _write_chunk_with_fallback(chunk_items[mid:])

        for start in range(0, len(items), chunk_size):
            await _write_chunk_with_fallback(items[start : start + chunk_size])
            if pause_seconds > 0 and start + chunk_size < len(items):
                await asyncio.sleep(pause_seconds)

        return len(writes)


# ---------------------------------------------------------------------------
# One-time load entry point (called from main.py at startup)
# ---------------------------------------------------------------------------

async def process_one_time_load(
    request: Request,
    dbname: str,
    collectionname: str,
    filter_key: str,
    filter_value: datetime | str,
    redis_client: Optional[object] = None,
    redis_prefix: str = "rt",
    *,
    batch_size: int = 100,
    concurrency: int = 10,
    redis_write_batch_size: int | None = None,
) -> int:
    """Stream Mongo documents for today, aggregate per conversationId, write to Redis.

    Returns the number of conversation groups processed.
    """
    logger = get_logger_from_request(request)
    cursor_batch_size = int(os.getenv("ONE_TIME_CURSOR_BATCH_SIZE", "8000"))
    convo_batch_size = int(os.getenv("ONE_TIME_CONVO_BATCH_SIZE", str(batch_size)))
    resolved_redis_write_batch_size = redis_write_batch_size or int(
        os.getenv("ONE_TIME_REDIS_WRITE_BATCH_SIZE", "250")
    )

    try:
        collection = await MongoDBConnectionPool.get_database_collection(
            dbname, collectionname
        )
    except Exception as exc:
        await logger.error(f"Failed to get collection {dbname}.{collectionname}: {exc}")
        return 0

    try:
        range_filter = _build_datetime_filter(filter_key, filter_value)
        if range_filter:
            start_dt = range_filter.get("$gte")
            end_dt = range_filter.get("$lte")
            filter_query = {filter_key: range_filter}
            if isinstance(start_dt, datetime) and isinstance(end_dt, datetime):
                filter_query = {
                    "$or": [
                        {filter_key: range_filter},
                        {
                            filter_key: {
                                "$gte": start_dt.isoformat(),
                                "$lte": end_dt.isoformat(),
                            }
                        },
                    ]
                }
            await logger.info(
                f"Range filter on {filter_key}: "
                f"{range_filter.get('$gte')} to {range_filter.get('$lte')}"
            )
        else:
            filter_query = {filter_key: filter_value}
    except Exception as exc:
        await logger.error(
            f"Failed to build filter for {filter_key}={filter_value}: {exc}"
        )
        return 0

    projection = {"_id": 0}
    cursor = collection.find(filter_query, projection).sort(
        [("conversationId", 1), ("createdDateTime", 1)]
    )
    cursor = cursor.batch_size(cursor_batch_size)

    metrics = RealtimeMetrics(
        redis_client=redis_client,
        logger=logger,
        redis_key_prefix=redis_prefix,
    )

    convo_batch: List[tuple] = []
    current_cid: Optional[str] = None
    current_docs: List[Dict[str, Any]] = []
    total_groups = 0

    async for raw_doc in cursor:
        try:
            doc = MongoDBConnectionPool.serialize(raw_doc)
        except Exception:
            doc = raw_doc
        cid = doc.get("conversationId") if isinstance(doc, dict) else None
        if not cid:
            continue

        if current_cid is None:
            current_cid = cid
            current_docs = [doc]
            continue

        if cid == current_cid:
            current_docs.append(doc)
        else:
            convo_batch.append((current_cid, current_docs))
            current_cid = cid
            current_docs = [doc]

            if len(convo_batch) >= convo_batch_size:
                processed = await metrics._process_initial_load_batches_of_conversations(
                    convo_batch,
                    concurrency=concurrency,
                    redis_write_batch_size=resolved_redis_write_batch_size,
                )
                total_groups += processed
                convo_batch = []

    if current_cid and current_docs:
        convo_batch.append((current_cid, current_docs))

    if convo_batch:
        processed = await metrics._process_initial_load_batches_of_conversations(
            convo_batch,
            concurrency=concurrency,
            redis_write_batch_size=resolved_redis_write_batch_size,
        )
        total_groups += processed

    await logger.info(
        f"One-time load complete: {total_groups} conversation groups "
        f"from {dbname}.{collectionname}"
    )
    return total_groups

_ISO_TS_PATTERN = re.compile(
    r"^(?P<year>\d{4})-(?P<month>\d{2})-(?P<day>\d{2})"
    r"(?:[T ](?P<hour>\d{2}):(?P<minute>\d{2})"
    r"(?::(?P<second>\d{2})(?:\.(?P<fraction>\d{1,6}))?)?"
    r"(?P<tz>Z|[+-]\d{2}:\d{2})?)?$"
)


def _parse_iso(ts: str | None):
    if ts is None:
        return None
    s = str(ts).strip()
    if not s:
        return None
    if s.endswith("Z"):
        s = s[:-1] + "+00:00"
    try:
        dt = datetime.fromisoformat(s)
        return dt if dt.tzinfo else dt.replace(tzinfo=timezone.utc)
    except Exception:
        return None


def _parse_iso_with_error(ts: str | None, field_name: str) -> tuple:
    if ts is None:
        return None, None

    raw = str(ts).strip()
    if not raw:
        return None, f"{field_name} cannot be empty"

    match = _ISO_TS_PATTERN.fullmatch(raw)
    if not match:
        return None, (
            f"{field_name} must be a valid ISO/RFC3339 datetime like 2026-03-15T00:00:00Z"
        )

    parts = match.groupdict()
    month = int(parts["month"])
    day = int(parts["day"])
    hour = int(parts["hour"] or 0)
    minute = int(parts["minute"] or 0)
    second = int(parts["second"] or 0)

    if not 1 <= month <= 12:
        return None, f"{field_name} has invalid month; must be 1-12"
    if not 1 <= day <= 31:
        return None, f"{field_name} has invalid day; must be 1-31"
    if not 0 <= hour <= 23:
        return None, f"{field_name} has invalid hour; must be 0-23"
    if not 0 <= minute <= 59:
        return None, f"{field_name} has invalid minute; must be 0-59"
    if not 0 <= second <= 59:
        return None, f"{field_name} has invalid second; must be 0-59"

    tz_part = parts.get("tz")
    if tz_part and tz_part != "Z":
        tz_hour = int(tz_part[1:3])
        tz_minute = int(tz_part[4:6])
        if not 0 <= tz_hour <= 23:
            return None, f"{field_name} has invalid timezone hour offset"
        if not 0 <= tz_minute <= 59:
            return None, f"{field_name} has invalid timezone minute offset"

    try:
        parsed = _parse_iso(raw)
    except Exception:
        parsed = None

    if parsed is None:
        return None, f"{field_name} has an invalid calendar date or time"

    return parsed, None


def _mongo_range_clause(field_name: str, start_dt: datetime, end_dt: datetime) -> dict:
    start_iso = start_dt.isoformat()
    end_iso = end_dt.isoformat()
    return {
        "$or": [
            {field_name: {"$gte": start_dt, "$lte": end_dt}},
            {field_name: {"$gte": start_iso, "$lte": end_iso}},
        ]
    }


def _validate_time_filters(
    start_time: str | None, end_time: str | None
) -> tuple:
    start_dt, start_error = _parse_iso_with_error(start_time, "start_time")
    if start_error:
        return None, None, start_error

    end_dt, end_error = _parse_iso_with_error(end_time, "end_time")
    if end_error:
        return None, None, end_error

    now_utc = datetime.now(timezone.utc)
    if start_dt and start_dt > now_utc:
        return None, None, "start_time cannot be in the future"

    if start_dt and end_dt and start_dt > end_dt:
        return None, None, "start_time must be less than or equal to end_time"

    return start_dt, end_dt, None


# How many keys to include in a single mget call.
# Matches the global default in redis_connect.py — overridable via env var.
_MGET_CHUNK = int(os.getenv("REDIS_MGET_CHUNK_SIZE", "200"))


@router.get(
    "/conversations",
    summary="List or filter aggregated conversations in Redis",
    tags=["Realtime"],
)
async def get_conversations(
    request: Request,
    prefix: str = "rt",
    limit: int | None = None,
    start_time: str | None = None,
    end_time: str | None = None,
    country: str | None = None,
    market: str | None = None,
):
    """Fetch aggregated conversation states stored in Redis under the given prefix.

    - Without filters: returns all (or up to `limit`) conversations.
    - With start_time / end_time: filters by `last_seen_at` (fallback `createdDateTime`).
    - With country / market: filters by those fields in the stored payload.
    """
    logger = get_logger_from_request(request)
    client = AWSRedis()
    normalized_prefix = prefix.rstrip(":")

    start_dt, end_dt, validation_error = _validate_time_filters(start_time, end_time)
    if validation_error:
        await logger.error(validation_error)
        return {"status": "error", "message": validation_error}

    filters_active = any([start_dt, end_dt, country, market])
    await logger.info(
        f"GET /conversations prefix={normalized_prefix} limit={limit} "
        f"filters_active={filters_active} start={start_dt} end={end_dt} "
        f"country={country} market={market}"
    )

    try:
        # ----------------------------------------------------------------
        # Fast path: limit requested, no filters
        # Scan up to `limit` keys and return immediately.
        # FIX: the previous code did a single mget(*all_keys).  We now
        # chunk it with _MGET_CHUNK so the call cannot time out when
        # the limit is large (e.g. 10 000).
        # ----------------------------------------------------------------
        if not filters_active and limit is not None:
            cfg = ConfigReader().get_redis_config()
            base_key = str(cfg.get("key"))
            full_prefix = f"{base_key}:{normalized_prefix}".rstrip(":") + ":"
            pattern = full_prefix + "*"

            keys: list[str] = []
            async for k in client._client.scan_iter(match=pattern, count=500):
                keys.append(k)
                if len(keys) >= limit:
                    break

            conversations: dict = {}
            if keys:
                # Chunked mget — safe regardless of how large `limit` is.
                for start in range(0, len(keys), _MGET_CHUNK):
                    chunk = keys[start : start + _MGET_CHUNK]
                    vals = await client._client.mget(*chunk)
                    for k, v in zip(chunk, vals):
                        try:
                            conversations[k] = json.loads(v) if v is not None else None
                        except Exception:
                            conversations[k] = v

            await logger.info(
                f"Fetched {len(conversations)} conversations (limit fast-path) "
                f"from Redis prefix {normalized_prefix}"
            )
            return {
                "prefix": normalized_prefix,
                "count": len(conversations),
                "conversations": conversations,
            }

        # ----------------------------------------------------------------
        # General path: load all keys under prefix.
        # get_all_with_prefix() already chunks internally (fixed in redis_connect.py).
        # ----------------------------------------------------------------
        data = await client.get_all_with_prefix(normalized_prefix)

        if not filters_active:
            await logger.info(
                f"Fetched {len(data)} conversations from Redis prefix {normalized_prefix}"
            )
            if limit is not None:
                limited_items = dict(list(data.items())[:limit])
                return {
                    "prefix": normalized_prefix,
                    "count": len(limited_items),
                    "conversations": limited_items,
                }
            return {
                "prefix": normalized_prefix,
                "count": len(data),
                "conversations": data,
            }

        # ----------------------------------------------------------------
        # Filter path
        # ----------------------------------------------------------------
        filtered: dict = {}
        for redis_key, payload in data.items():
            if not isinstance(payload, dict):
                continue

            if start_dt or end_dt:
                ts_candidate = payload.get("last_seen_at") or payload.get("createdDateTime")
                ts_val = _parse_iso(ts_candidate)
                if start_dt and (ts_val is None or ts_val < start_dt):
                    continue
                if end_dt and (ts_val is None or ts_val > end_dt):
                    continue

            if market and str(payload.get("market") or "").lower() != market.lower():
                continue

            if country:
                country_candidates = [
                    payload.get("country"),
                    payload.get("region"),
                    payload.get("market"),
                ]
                if not any(
                    str(c or "").lower() == country.lower() for c in country_candidates
                ):
                    continue

            filtered[redis_key] = payload
            if limit is not None and len(filtered) >= limit:
                break

        await logger.info(
            f"Filtered prefix={normalized_prefix} start={start_dt} end={end_dt} "
            f"country={country} market={market} limit={limit} matched={len(filtered)}"
        )
        return {
            "prefix": normalized_prefix,
            "count": len(filtered),
            "filters": {

@router.get(
    "/temp/redis_diagnostics",
    summary="TEMP: Redis connectivity + key sample",
    tags=["Diagnostics"],
)
async def temp_redis_diagnostics(
    request: Request,
    prefix: str = "rt",
    sample_limit: int = 20,
):
    """Return ping result and a small key sample — useful for debugging Redis connectivity."""
    logger = get_logger_from_request(request)
    cfg = ConfigReader().get_redis_config()
    client = AWSRedis()
    normalized_prefix = prefix.rstrip(":") + ":"
    full_prefix = str(cfg.get("key")) + ":" + normalized_prefix
    full_prefix = full_prefix.rstrip(":") + ":"
    pattern = full_prefix + "*"

    try:
        ping_ok = await client._client.ping()
    except Exception as exc:
        ping_ok = False
        await logger.error(f"Redis ping failed: {exc}")

    keys: list[str] = []
    try:
        async for k in client._client.scan_iter(match=pattern, count=500):
            keys.append(k)
            if len(keys) >= sample_limit:
                break
    except Exception as exc:
        await logger.error(f"Redis scan_iter failed: {exc}")
        return {
            "status": "error",
            "message": str(exc),
            "host": cfg.get("host"),
            "port": cfg.get("port"),
            "prefix": normalized_prefix,
        }

    sample_values: dict = {}
    try:
        if keys:
            vals = await client._client.mget(*keys)
            for k, v in zip(keys, vals):
                sample_values[k] = v
    except Exception as exc:
        await logger.error(f"Redis mget failed: {exc}")

    return {
        "status": "ok" if ping_ok else "degraded",
        "host": cfg.get("host"),
        "port": cfg.get("port"),
        "prefix": normalized_prefix,
        "base_key": cfg.get("key"),
        "pattern": pattern,
        "ping": ping_ok,
        "sample_count": len(keys),
        "sample_keys": keys,
        "sample_values": sample_values,
    }


@router.delete(
    "/temp/redis_conversations",
    summary="TEMP: Delete all conversation data in Redis for a prefix",
    tags=["Diagnostics"],
)
async def temp_delete_redis_conversations(
    request: Request,
    prefix: str = "rt",
):
    """Remove all Redis keys under base_key:prefix:*.  Use with care."""
    logger = get_logger_from_request(request)
    cfg = ConfigReader().get_redis_config()
    client = AWSRedis()

    normalized_prefix = prefix.rstrip(":")
    base_key = str(cfg.get("key"))
    full_prefix = f"{base_key}:{normalized_prefix}".rstrip(":") + ":"
    pattern = full_prefix + "*"
    deleted_count = 0
    scanned_count = 0

    try:
        batch: list[str] = []
        async for key in client._client.scan_iter(match=pattern, count=500):
            scanned_count += 1
            batch.append(key)
            if len(batch) >= 500:
                deleted_count += await client.delete_keys(batch)
                batch = []
        if batch:
            deleted_count += await client.delete_keys(batch)

        await logger.warning(
            f"Redis purge: removed {deleted_count} key(s) for pattern {pattern} "
            f"(scanned={scanned_count})"
        )
        return {
            "status": "ok",
            "message": "Redis conversation purge completed",
            "base_key": base_key,
            "prefix": normalized_prefix,
            "pattern": pattern,
            "scanned_count": scanned_count,
            "deleted_count": deleted_count,
        }
    except Exception as exc:
        await logger.error(f"Redis purge failed for pattern {pattern}: {exc}")
        return {
            "status": "error",
            "message": str(exc),
            "base_key": base_key,
            "prefix": normalized_prefix,
            "pattern": pattern,
            "scanned_count": scanned_count,
            "deleted_count": deleted_count,
        }


@router.get(
    "/mongo/conversations",
    summary="Fetch dumped conversations from Mongo by time range",
    tags=["Realtime"],
)
async def get_dumped_conversations(
    request: Request,
    start_time: str,
    end_time: str,
    limit: int | None = None,
    dbname: str | None = None,
    collection: str | None = None,
):
    """Fetch aggregated conversation documents from the nightly Redis-dump collection in Mongo."""
    logger = get_logger_from_request(request)

    start_dt, end_dt, validation_error = _validate_time_filters(start_time, end_time)
    if validation_error:
        await logger.error(validation_error)
        return {"status": "error", "message": validation_error}

    mongo_cfg = ConfigReader().get_mongo_config()
    target_db = dbname or mongo_cfg.get("database_name") or "ccaas"
    target_collection = collection or os.getenv(
        "REDIS_DUMP_COLLECTION", "conversation_anirudh_download2"
    )

    query = {
        "$or": [
            _mongo_range_clause("first_seen_at", start_dt, end_dt),
            _mongo_range_clause("last_seen_at", start_dt, end_dt),
        ]
    }
    projection = {"_id": 0}

    await logger.info(
        f"Fetching dumped conversations from {target_db}.{target_collection} "
        f"start={start_dt.isoformat()} end={end_dt.isoformat()} limit={limit}"
    )

    try:
        collection_handle = await MongoDBConnectionPool.get_database_collection(
            target_db, target_collection
        )
        cursor = collection_handle.find(query, projection).sort(
            [("first_seen_at", 1), ("last_seen_at", 1)]
        )
        if limit is not None and limit > 0:
            cursor = cursor.limit(int(limit))

        conversations = []
        async for raw_doc in cursor:
            try:
                doc = MongoDBConnectionPool.serialize(raw_doc)
            except Exception:
                doc = raw_doc
            conversations.append(doc)

        return {
            "status": "ok",
            "database": target_db,
            "collection": target_collection,
            "count": len(conversations),
            "filters": {
                "start_time": start_time,
                "end_time": end_time,
                "limit": limit,
            },
            "conversations": conversations,
        }
    except Exception as exc:
        await logger.error(f"Failed to fetch dumped conversations from Mongo: {exc}")
        return {
            "status": "error",
            "message": str(exc),
            "database": target_db,
            "collection": target_collection,
        }
