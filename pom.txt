# This file handles most of the Mongo-side infrastructure for the service.
#
# It is responsible for:
# 1. creating and reusing one shared Mongo client for the whole app process
# 2. reading documents from Mongo in API and batch flows
# 3. listening to Mongo change streams (CDC) and keeping Redis updated
# 4. saving Redis snapshots back into Mongo before Redis keys expire at midnight UTC
#
# A simple way to think about it:
# - `realtime_metrics.py` decides WHAT a conversation means
# - `connectionpool.py` decides HOW data moves between Mongo, Redis, and the app

# Shared error strings used when a Mongo-related API call fails.
INTERNAL_SERVER_ERR = "Internal Server Error"
INTERNAL_SERVER_MSG = "An unexpected error occurred. Please try again later."


def get_logger_from_request(request: Request):
    # The app stores the logger in `request.app.state`, so this helper keeps access simple.
    return request.app.state.logger


def _parse_date_value(raw_value: str) -> datetime | None:
    """Try to parse a date/datetime string into a timezone-aware datetime (UTC)."""
    # This helper accepts both plain dates and full ISO datetimes so callers do not
    # need to worry about low-level parsing rules.
    if not raw_value:
        return None
    value = raw_value.strip()
    if not value:
        return None

    # Support a few human-friendly date-only formats, not just one strict format.
    for fmt in ("%Y-%m-%d", "%d/%m/%Y", "%m/%d/%Y"):
        try:
            dt = datetime.strptime(value, fmt)
            return dt.replace(tzinfo=timezone.utc)
        except Exception:
            pass

    # If it was not a plain date, try ISO-style datetime parsing.
    try:
        iso_value = value[:-1] + "+00:00" if value.endswith("Z") else value
        dt = datetime.fromisoformat(iso_value)
        return dt if dt.tzinfo else dt.replace(tzinfo=timezone.utc)
    except Exception:
        return None


def _build_datetime_filter(key_name: str, raw_value):
    """
    Build a Mongo range filter for datetime fields.

    - Accepts date-only inputs (e.g., "06/02/2026") and expands to start-of-day UTC through now.
    - Accepts datetime inputs and uses them as the lower bound through now.

    Returns None if parsing fails so the caller can fall back to equality filters.
    """
    # Only build a datetime range when the field itself looks like a datetime field.
    if not key_name or "date" not in key_name.lower():
        return None

    parsed: datetime | None = None
    date_only = False

    if isinstance(raw_value, datetime):
        parsed = raw_value if raw_value.tzinfo else raw_value.replace(tzinfo=timezone.utc)
        # If the caller gave midnight exactly, we treat that as a full-day filter.
        date_only = parsed.timetz() == time(0, 0)
    elif isinstance(raw_value, str):
        parsed = _parse_date_value(raw_value)
        # Strings without a time portion are also treated as full-day filters.
        date_only = "T" not in raw_value and len(raw_value.strip()) <= 10

    if not parsed:
        return None

    # For date-only input, start at the beginning of that UTC day.
    # For full datetime input, use the provided value directly.
    start = parsed.replace(hour=0, minute=0, second=0, microsecond=0) if date_only else parsed
    now_utc = datetime.now(timezone.utc)
    # The upper bound is always "now" because these queries are designed for recent data fetches.
    return {"$gte": start, "$lte": now_utc}


class MongoDBConnectionPool:
    # This class behaves like an async singleton.
    # In other words, one app process should create one shared Mongo client,
    # and the rest of the code should reuse it.

    @classmethod
    async def close_pool(cls):
        """
        Close the MongoDB connection pool if it exists.
        """
        if cls._client is not None:
            # Closing the client also closes the underlying network connections.
            cls._client.close()
            cls._client = None

    """
    Async Singleton MongoDB connection pool for FastAPI apps.
    - Call await MongoDBConnectionPool.init_pool(config) ONCE at app startup (e.g. in lifespan/startup event).
    - Use await MongoDBConnectionPool.get_client() or get_collection() anywhere in the app.
    This avoids repeated config reads and ensures only one AsyncIOMotorClient is ever created.
    """
    _client = None
    _config = None
    _lock = asyncio.Lock()

    @staticmethod
    def serialize(document):
        """
        Recursively convert datetime objects to ISO format in a MongoDB document or list of documents.
        :param document: A single document (dict) or list of documents (list of dicts).
        :return: Document(s) with datetime fields serialized to ISO format.
        """
        # Many parts of the service expect plain JSON-friendly objects.
        # This helper walks the returned Mongo structure and converts datetime objects
        # into ISO strings so they can be logged, returned, or further processed safely.
        if isinstance(document, list):
            return [MongoDBConnectionPool.serialize(doc) for doc in document]
        elif isinstance(document, dict):
            # `_id` is removed because the rest of this service usually does not need it,
            # and removing it keeps the downstream shape simpler and more consistent.
            doc = dict(document)
            doc.pop('_id', None)
            for key, value in doc.items():
                if isinstance(value, datetime):
                    doc[key] = value.isoformat()
                elif isinstance(value, dict):
                    doc[key] = MongoDBConnectionPool.serialize(value)
                elif isinstance(value, list):
                    doc[key] = MongoDBConnectionPool.serialize(value)
            return doc
        else:
            return document

    @classmethod
    async def init_pool(cls, config=None):
        """
        Create the shared Mongo client once.

        Why this matters in the codebase:
        - `main.py` calls this at startup.
        - every later Mongo read depends on this having succeeded.
        - using one shared client is much cheaper than making a new client per request.
        """
        async with cls._lock:
            if cls._client is None:
                if config is None:
                    # Read config lazily if the caller did not explicitly provide one.
                    config = ConfigReader().get_mongo_config()
                cls._config = config
                cls._client = AsyncIOMotorClient(
                    config['uri'],
                    maxPoolSize=config['max_pool_size'],
                    minPoolSize=config['min_pool_size'],
                    maxIdleTimeMS=config['idle_timeout_ms'],
                    appname='prompt_manager'
                )

    @classmethod
    async def get_client(cls):
        # If this raises, it usually means startup did not initialize Mongo correctly.
        if cls._client is None:
            raise RuntimeError("MongoDBConnectionPool not initialized. Call init_pool() at startup.")
        return cls._client

    @classmethod
    async def get_database_collection(cls, dbname, collectionname):
        """
        Return a Motor collection object for the given database and collection.
        """
        # This hides the low-level client/database indexing from the rest of the code.
        client = await cls.get_client()
        db = client[dbname]
        return db[collectionname]

    @classmethod
    async def get_documents(cls, request: Request, dbname, collectionname, key, key_name):
        """
        Fetch all documents from a specific collection that match the provided filter.
        :param dbname: Name of the database
        :param collectionname: Name of the collection
        :param key: Value to filter on
        :param key_name: Column / field name to filter on
        :return: List of serialized documents or "No data Available"
        """
        # This is a direct query helper used by API-facing flows.
        # It supports both simple equality filters and datetime range filters.
        logger = get_logger_from_request(request)
        try:
            collection = await cls.get_database_collection(dbname, collectionname)
            range_filter = _build_datetime_filter(key_name, key)
            if range_filter:
                # If the target field looks like a datetime field, build a time range query.
                filter_query = {key_name: range_filter}

                # Some collections store dates as real Mongo datetimes.
                # Some store them as strings.
                # We query both forms to be more tolerant of mixed historical data.
                start_dt = range_filter.get("$gte")
                end_dt = range_filter.get("$lte")
                if isinstance(start_dt, datetime) and isinstance(end_dt, datetime):
                    start_iso = start_dt.isoformat()
                    end_iso = end_dt.isoformat()
                    filter_query = {
                        "$or": [
                            {key_name: range_filter},
                            {key_name: {"$gte": start_iso, "$lte": end_iso}},
                        ]
                    }

                await logger.info(
                    f"Applying range filter on {key_name}: {range_filter.get('$gte')} to {range_filter.get('$lte')}"
                )
            else:
                # If no datetime filter applies, fall back to a normal exact match query.
                filter_query = {key_name: key}
            projection = {"_id": 0}
            cursor = collection.find(filter_query, projection)
            documents = []
            async for doc in cursor:
                # This helper is intended for direct reads, so collecting all docs is fine here.
                documents.append(doc)
            if documents:
                return cls.serialize(documents)
            else:
                return "No data Available"
        except Exception as e:
            # Return a stable JSON error payload instead of letting the exception leak directly.
            await logger.error(f"Failed to retrieve documents from collection '{collectionname}': {e}")
            return JSONResponse(
                status_code=500,
                content={
                    "error": INTERNAL_SERVER_ERR,
                    "message": INTERNAL_SERVER_MSG,
                    "statusCodeValue": 500
                }
            )

    @classmethod
    async def get_documents_grouped_by_conversation_ids(
        cls,
        request: Request,
        dbname,
        collectionname,
        conversation_ids: list[str],
    ) -> dict[str, list[dict]]:
        """Fetch all docs for the given conversation IDs in one query and group them in memory."""
        # This helper is very important for CDC correctness.
        # When a single Mongo document changes, the aggregator does not just process that one doc.
        # Instead, it refetches the full conversation so Redis is rebuilt from complete truth.
        if not conversation_ids:
            return {}

        logger = get_logger_from_request(request)
        try:
            collection = await cls.get_database_collection(dbname, collectionname)
            projection = {"_id": 0}
            fetch_batch_size = int(os.getenv("CDC_FETCH_CURSOR_BATCH_SIZE", "10000"))
            # Sorting here gives a stable and predictable document order.
            cursor = collection.find(
                {"conversationId": {"$in": list(conversation_ids)}},
                projection,
            ).sort([("conversationId", 1), ("createdDateTime", 1)])
            cursor = cursor.batch_size(fetch_batch_size)

            # Pre-seed the output so every requested ID exists in the result map.
            grouped_docs: dict[str, list[dict]] = {cid: [] for cid in conversation_ids}
            async for raw_doc in cursor:
                try:
                    doc = cls.serialize(raw_doc)
                except Exception:
                    doc = raw_doc

                if not isinstance(doc, dict):
                    continue

                cid = doc.get("conversationId")
                if not cid:
                    # Skip anything that cannot be tied back to a conversation group.
                    continue
                grouped_docs.setdefault(cid, []).append(doc)

            return grouped_docs
        except Exception as e:
            await logger.error(
                f"Failed to retrieve grouped conversation documents from collection '{collectionname}': {e}"
            )
            return {}

    @classmethod
    async def start_cdc_stream(
        cls,
        app,
        *,
        dbname: str | None = None,
        collectionname: str = "anirudh_conversation_test2",
        redis_prefix: str = "rt",
        max_concurrency: int = 40,
        retry_delay_seconds: int = 5,
    ):
        """Begin a change-stream listener that mirrors Mongo updates into Redis in near real time.

        - Waits for insert/update/replace operations on the target collection.
        - For each change, refetches the full conversationId group and re-runs aggregation logic.
        - Writes/updates the Redis record (with TTL to next midnight handled by Redis client).
        - On errors, logs and retries with a short backoff to avoid tight loops.
        """

        logger = getattr(app.state, "logger", None)

        class _RequestProxy:
            """Lightweight shim so existing get_documents signature can log via app logger."""
            def __init__(self, app_ref):
                self.app = app_ref

        request_proxy = _RequestProxy(app)

        async def _log(level_fn, msg: str):
            # Safe logger wrapper: logging should never crash the CDC loop itself.
            try:
                if level_fn:
                    await level_fn(msg)
            except Exception:
                pass

        while True:
            # Outer restart loop.
            # If the stream dies for any reason, we log, sleep briefly, and rebuild it.
            try:
                target_db = dbname or (cls._config.get("database_name") if cls._config else None)
                if not target_db:
                    config = ConfigReader().get_mongo_config()
                    target_db = config.get("database_name")

                collection = await cls.get_database_collection(target_db, collectionname)
                pipeline = [{"$match": {"operationType": {"$in": ["insert", "update", "replace"]}}}]

                await _log(logger.info if logger else None, f"CDC stream: watching {target_db}.{collectionname} for changes")

                from app.core.redis_connect import AWSRedis
                from app.service.realtime_metrics import RealtimeMetrics

                redis_client = AWSRedis()
                metrics = RealtimeMetrics(
                    redis_client=redis_client,
                    logger=logger,
                    redis_key_prefix=redis_prefix,
                )

                # CDC_QUEUE_MAXSIZE bounds memory under sustained high load.
                # At 3k docs/second the queue fills fast — without a cap, memory
                # grows unboundedly during a busy day. With a cap, the change-stream
                # reader applies back-pressure (waits) instead of queuing forever.
                # The pending/dirty dedup means a full queue does not mean data loss —
                # IDs already in pending are marked dirty and reprocessed, not dropped.
                queue_maxsize = int(os.getenv("CDC_QUEUE_MAXSIZE", "20000"))
                queue: asyncio.Queue[str] = asyncio.Queue(maxsize=queue_maxsize)
                pending: set[str] = set()
                dirty: set[str] = set()
                state_lock = asyncio.Lock()

                batch_window_ms = int(os.getenv("CDC_BATCH_WINDOW_MS", "1500"))
                batch_window = max(0.05, batch_window_ms / 1000)
                sem = asyncio.Semaphore(max(1, max_concurrency))
                max_batches_in_flight = int(os.getenv("CDC_MAX_BATCHES_IN_FLIGHT", "3"))
                batch_sem = asyncio.Semaphore(max(1, max_batches_in_flight))
                # CDC_NUM_CONSUMERS controls how many parallel batch consumers drain
                # the queue. One consumer cannot keep up at sustained 3k docs/second.
                # Multiple consumers drain the queue faster, keeping memory stable
                # and processing latency low during busy periods.
                num_consumers = int(os.getenv("CDC_NUM_CONSUMERS", "3"))
                active_batch_tasks: set[asyncio.Task] = set()

                # FIX: CDC process_batch is now chunked to handle large spikes safely.
                #
                # Previously, if a burst of 40-50k documents arrived and the debounce
                # window grouped them into one large batch, two problems occurred:
                # 1. get_documents_grouped_by_conversation_ids was called with a huge
                #    $in list, putting heavy pressure on Mongo in one shot.
                # 2. asyncio.gather(*tasks) created one task per conversation ID in the
                #    entire batch simultaneously, causing memory pressure under large spikes.
                #
                # The fix splits the batch into chunks of CDC_PROCESS_CHUNK_SIZE (default 200)
                # before hitting Mongo or creating tasks. Each chunk is processed sequentially,
                # keeping Mongo $in queries bounded and task count manageable.
                # The dirty/requeue mechanism is unchanged so no change events are lost.
                async def process_batch(batch: set[str]):
                    cdc_chunk_size = int(os.getenv("CDC_PROCESS_CHUNK_SIZE", "200"))
                    batch_list = list(batch)

                    if logger:
                        try:
                            await logger.debug(
                                f"CDC processing batch of {len(batch_list)} conversation(s) "
                                f"in chunks of {cdc_chunk_size}"
                            )
                        except Exception:
                            pass

                    for i in range(0, len(batch_list), cdc_chunk_size):
                        chunk = set(batch_list[i:i + cdc_chunk_size])

                        # Bounded Mongo $in query — at most cdc_chunk_size IDs per query.
                        docs_by_cid = await cls.get_documents_grouped_by_conversation_ids(
                            request_proxy,
                            target_db,
                            collectionname,
                            list(chunk),
                        )
                        # Bounded Redis batch read — also chunked inside get_interactions_batch.
                        existing_by_cid = await redis_client.get_interactions_batch(
                            list(chunk), key_prefix=redis_prefix
                        )
                        writes: dict[str, dict] = {}

                        async def handle(cid: str):
                            async with sem:
                                try:
                                    docs = docs_by_cid.get(cid) or []
                                    if not docs:
                                        if logger:
                                            try:
                                                await logger.debug(f"CDC: no docs fetched for cid={cid}")
                                            except Exception:
                                                pass
                                        return
                                    if logger:
                                        try:
                                            await logger.debug(f"CDC: fetched {len(docs)} docs for cid={cid}")
                                        except Exception:
                                            pass
                                    aggregated = metrics._build_aggregated_payload(docs)
                                    if not aggregated:
                                        return
                                    merged = metrics._merge_with_existing(existing_by_cid.get(cid), aggregated)
                                    writes[cid] = merged
                                except Exception as w_exc:
                                    await _log(logger.error if logger else None, f"CDC worker error for {cid}: {w_exc}")
                                finally:
                                    requeue = False
                                    async with state_lock:
                                        if cid in dirty:
                                            dirty.discard(cid)
                                            requeue = True
                                        else:
                                            pending.discard(cid)

                                    if requeue:
                                        await queue.put(cid)

                        # At most cdc_chunk_size tasks created at once — safe under any spike.
                        tasks = [asyncio.create_task(handle(cid)) for cid in chunk]
                        if tasks:
                            await asyncio.gather(*tasks)
                        if writes:
                            # Chunked inside write_interactions_batch — safe for large writes.
                            await redis_client.write_interactions_batch(writes, key_prefix=redis_prefix)

                async def batch_consumer():
                    # This background consumer keeps pulling conversation IDs from the queue,
                    # waits a very short time for nearby arrivals, and then launches a batch.
                    while True:
                        cid = await queue.get()
                        batch = {cid}
                        queue.task_done()
                        start = time_module.monotonic()
                        while True:
                            remaining = batch_window - (time_module.monotonic() - start)
                            if remaining <= 0:
                                break
                            try:
                                nxt = await asyncio.wait_for(queue.get(), timeout=remaining)
                                batch.add(nxt)
                                queue.task_done()
                            except asyncio.TimeoutError:
                                break

                        async def _run_batch(batch_to_process: set[str]):
                            async with batch_sem:
                                await process_batch(batch_to_process)

                        batch_task = asyncio.create_task(_run_batch(batch))
                        active_batch_tasks.add(batch_task)
                        batch_task.add_done_callback(active_batch_tasks.discard)

                # Start multiple consumers so the queue drains faster under
                # sustained high load. Each consumer independently pulls batches
                # and processes them — they share the same pending/dirty state
                # safely via state_lock.
                consumer_tasks = [
                    asyncio.create_task(batch_consumer())
                    for _ in range(num_consumers)
                ]
                await _log(
                    logger.info if logger else None,
                    f"CDC batch consumers started (count={num_consumers}, window={batch_window_ms}ms, "
                    f"max_batches_in_flight={max_batches_in_flight}, queue_maxsize={queue_maxsize}) "
                    f"for {target_db}.{collectionname}",
                )

                async with collection.watch(pipeline, full_document="updateLookup") as stream:
                    async for change in stream:
                        full_doc = change.get("fullDocument") if isinstance(change, dict) else None
                        if not full_doc:
                            continue
                        cid = full_doc.get("conversationId") if isinstance(full_doc, dict) else None
                        if not cid:
                            continue
                        if logger:
                            try:
                                await logger.debug(f"CDC change captured for cid={cid}")
                            except Exception:
                                pass
                        async with state_lock:
                            if cid in pending:
                                # Already queued — mark dirty so it gets reprocessed
                                # after the current run. This is the primary dedup
                                # mechanism and means a full queue never loses data
                                # for already-tracked conversations.
                                dirty.add(cid)
                                continue
                            pending.add(cid)

                        # Back-pressure: if the queue is full, wait here instead of
                        # growing memory unboundedly. The change stream reader pauses
                        # naturally — Mongo buffers changes on its side during the
                        # wait so no events are lost at the source.
                        await queue.put(cid)

            except (PyMongoError, Exception) as exc:
                await _log(logger.error if logger else None, f"CDC stream error for {collectionname}: {exc}")
                await asyncio.sleep(max(1, retry_delay_seconds))
            finally:
                # Cancel all consumer tasks from this session on restart.
                try:
                    for ct in locals().get("consumer_tasks", []):
                        ct.cancel()
                except Exception:
                    pass
                try:
                    batch_tasks = list(locals().get("active_batch_tasks", set()))
                    for task in batch_tasks:
                        task.cancel()
                except Exception:
                    pass

    @classmethod
    async def dump_redis_prefix_to_mongo(
        cls,
        app,
        *,
        dbname: str | None = None,
        redis_prefix: str = "rt",
        target_collection: str = "conversation_anirudh_download2",
        ttl_extension_seconds: int = 900,
        delete_after_dump: bool = True,
        batch_size: int = 1000,
    ):
        """Snapshot Redis prefix data into Mongo (run around midnight before TTL expiry)."""
        # Redis is intentionally ephemeral in this system.
        # This dump function preserves end-of-day snapshots in Mongo so historical lookup remains possible.

        logger = getattr(app.state, "logger", None)

        def _now():
            return datetime.now(timezone.utc)

        try:
            from app.core.redis_connect import AWSRedis

            redis_client = AWSRedis()
            # Extend TTLs while reading to reduce the risk of keys expiring during the dump itself.
            data = await redis_client.get_all_with_prefix(
                redis_prefix,
                extend_ttl_seconds=ttl_extension_seconds,
                delete_after=False,
            )

            if not data:
                if logger:
                    await logger.info(f"Redis dump: no keys found for prefix {redis_prefix}")
                return 0

            target_db = dbname or (cls._config.get("database_name") if cls._config else None)
            if not target_db:
                config = ConfigReader().get_mongo_config()
                target_db = config.get("database_name")

            collection = await cls.get_database_collection(target_db, target_collection)

            docs = []
            dumped_at = _now()
            for full_key, payload in data.items():
                doc = payload if isinstance(payload, dict) else {"value": payload}
                doc = dict(doc)
                doc["_redis_key"] = full_key
                doc["_dumped_at_utc"] = dumped_at
                docs.append(doc)

            # Insert in smaller chunks so one oversized write does not affect the whole dump.
            inserted = 0
            for i in range(0, len(docs), batch_size):
                batch = docs[i: i + batch_size]
                try:
                    res = await collection.insert_many(batch, ordered=False)
                    inserted += len(res.inserted_ids)
                except Exception as batch_exc:
                    if logger:
                        await logger.error(f"Redis dump batch insert failed: {batch_exc}")

            if delete_after_dump:
                try:
                    # Only remove Redis keys after the Mongo copy step has finished.
                    await redis_client.delete_keys(list(data.keys()))
                except Exception as del_exc:
                    if logger:
                        await logger.error(f"Redis dump cleanup failed: {del_exc}")

            if logger:
                await logger.info(
                    f"Redis dump complete: {inserted} docs from prefix {redis_prefix} into {target_db}.{target_collection}"
                )

            return inserted
        except Exception as exc:
            if logger:
                await logger.error(f"Redis dump failed for prefix {redis_prefix}: {exc}")
            return 0

    @classmethod
    async def schedule_daily_redis_dump(
        cls,
        app,
        *,
        redis_prefix: str = "rt",
        target_collection: str = "conversation_anirudh_download2",
        lead_seconds: int = 30,
        ttl_extension_seconds: int = 900,
    ):
        """Run a dump near midnight UTC every day, with a small lead to beat TTL expiry."""
        # This scheduler is the bridge from "today's realtime Redis state" to
        # "historical Mongo snapshot data".
        # Without it, Redis expiry would erase the aggregated state at midnight.

        logger = getattr(app.state, "logger", None)

        def seconds_until_next_midnight_utc():
            now = datetime.now(timezone.utc)
            next_midnight = (now + timedelta(days=1)).replace(hour=0, minute=0, second=0, microsecond=0)
            return max(1, int((next_midnight - now).total_seconds()))

        while True:
            try:
                sleep_for = max(1, seconds_until_next_midnight_utc() - max(0, lead_seconds))
                if logger:
                    await logger.info(
                        f"Redis dump scheduler sleeping {sleep_for}s before next run (prefix={redis_prefix}, collection={target_collection})"
                    )
                await asyncio.sleep(sleep_for)

                if logger:
                    await logger.info(
                        f"Starting scheduled Redis dump (prefix={redis_prefix}, collection={target_collection})"
                    )
                await cls.dump_redis_prefix_to_mongo(
                    app,
                    redis_prefix=redis_prefix,
                    target_collection=target_collection,
                    ttl_extension_seconds=ttl_extension_seconds,
                    delete_after_dump=True,
                )
            except Exception as exc:
                if logger:
                    await logger.error(f"Scheduled Redis dump failed: {exc}")
            finally:
                # Small buffer prevents an accidental tight loop if timing lands awkwardly.
                await asyncio.sleep(5)
