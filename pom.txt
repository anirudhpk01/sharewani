async def get_all_with_prefix(
        self,
        prefix: str,
        *,
        extend_ttl_seconds: int | None = None,
        delete_after: bool = False,
        raise_on_error: bool = False,
    ) -> dict:
        read_timeout = float(os.getenv("REDIS_READ_TOTAL_TIMEOUT_SECONDS", "8"))
        try:
            return await asyncio.wait_for(
                self._get_all_with_prefix_inner(
                    prefix,
                    extend_ttl_seconds=extend_ttl_seconds,
                    delete_after=delete_after,
                    raise_on_error=raise_on_error,
                ),
                timeout=read_timeout,
            )
        except asyncio.TimeoutError:
            logger.error(
                "get_all_with_prefix timed out after %.1fs for prefix %s",
                read_timeout,
                prefix,
            )
            if raise_on_error:
                raise
            return {}

    async def _get_all_with_prefix_inner(
        self,
        prefix: str,
        *,
        extend_ttl_seconds: int | None = None,
        delete_after: bool = False,
        raise_on_error: bool = False,
    ) -> dict:
        """Inner implementation of get_all_with_prefix — wrapped with timeout by caller."""
        p = self._build_prefix(prefix)
        pattern = p + "*"
        scan_count = max(1, _env_int("REDIS_SCAN_COUNT", 200))
        fetch_chunk_size = max(1, _env_int("REDIS_FETCH_CHUNK_SIZE", 200))

        try:
            keys = []
            async for k in self._client.scan_iter(match=pattern, count=scan_count):
                keys.append(k)

            if not keys:
                return {}

            if extend_ttl_seconds and extend_ttl_seconds > 0:
                await self._expire_keys_in_chunks(self._client, keys, int(extend_ttl_seconds), fetch_chunk_size)

            result = {}
            for start in range(0, len(keys), fetch_chunk_size):
                key_chunk = keys[start : start + fetch_chunk_size]
                vals = await self._client.mget(*key_chunk)
                for k, v in zip(key_chunk, vals):
                    try:
                        parsed = json.loads(v) if v is not None else None
                    except Exception:
                        parsed = v
                    result[k] = parsed

            if delete_after:
                try:
                    for start in range(0, len(keys), fetch_chunk_size):
                        await self._client.delete(*keys[start : start + fetch_chunk_size])
                except AuthenticationError:
                    for start in range(0, len(keys), fetch_chunk_size):
                        await self._client.delete(*keys[start : start + fetch_chunk_size])

            return result
        except Exception as e:
            logger.error(f"Failed to fetch keys with prefix {p}: {e}")
            if raise_on_error:
                raise
            return {}
